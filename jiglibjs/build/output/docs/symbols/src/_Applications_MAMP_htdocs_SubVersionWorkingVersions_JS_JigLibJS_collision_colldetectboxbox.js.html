 62 	/**
 63 	 * @function disjoint tests for disjoint or intersection
 64 	 * @param {SpanData} out the SpanData object to apply test results to
 65 	 * @param {array} axis the axis expressed as a 3D vector
 66 	 * @param {JBox} box0 the first box to use for testing
 67 	 * @param {JBox} box1 the second box to use for testing
 68 	 * @returns true if disjoint, false if intersecting
 69 	 * @type boolean
 70 	 **/
 71 	CollDetectBoxBox.prototype.disjoint=function(out, axis, box0, box1){
 72 		var obj0 = box0.getSpan(axis);
 73 		var obj1 = box1.getSpan(axis);
 74 		var obj0Min = obj0.min;
 75 		var obj0Max = obj0.max;
 76 		var obj1Min = obj1.min;
 77 		var obj1Max = obj1.max;
 78 		var mmin = Math.min;
 79 		
 80 		if (obj0Min > (obj1Max + JConfig.collToll + JNumber3D.NUM_TINY) || obj1Min > (obj0Max + JConfig.collToll + JNumber3D.NUM_TINY)){
 81 			out.flag = true;
 82 			return true;
 83 		}
 84 		if ((obj0Max > obj1Max) && (obj1Min > obj0Min)){
 85 			out.depth = mmin(obj0Max - obj1Min, obj1Max - obj0Min);
 86 		}else if ((obj1Max > obj0Max) && (obj0Min > obj1Min)){
 87 			out.depth = mmin(obj1Max - obj0Min, obj0Max - obj1Min);
 88 		}else{
 89 			out.depth = (obj0Max < obj1Max) ? obj0Max : obj1Max;
 90 			out.depth -= (obj0Min > obj1Min) ? obj0Min : obj1Min;
 91 		}
 92 		out.flag = false;
 93 		return false;
 94 	};

 97 	/**
 98 	 * @function addPoint conditionally adds one 3D vector to a collection of other 3D vectors
 99 	 * @param {array} contactPoints a collection of points (3D vectors)
100 	 * @param {array} pt the point to add expressed as a 3D vector
101 	 * @param {number} combinationDistanceSq the maximum length squared allowed between pt and any of contactPoints
102 	 * @returns true if pt was added to contactPoints, false if combinationDistanceSq was ever exceeded
103 	 * @type boolean
104 	 **/
105 	CollDetectBoxBox.prototype.addPoint=function(contactPoints, pt, combinationDistanceSq){
106 		for(var i=0,cpsl=contactPoints.length;i<cpsl;i++){
107 			var contactPoint=contactPoints[i];
108 			if (Vector3DUtil.get_lengthSquared(Vector3DUtil.subtract(contactPoint, pt)) < combinationDistanceSq){
109 				contactPoint = JNumber3D.getDivideVector(Vector3DUtil.add(contactPoint, pt), 2);
110 				return false;
111 			}
112 		}
113 		contactPoints.push(pt);
114 		return true;
115 	};

118 	/**
119 	 * @function getSupportPoint
120 	 * @param {JBox} box
121 	 * @param {array} axis the axis expressed as a 3D vector
122 	 * @returns the point expressed as a 3D Vector
123 	 * @type {array}
124 	 **/
125 	CollDetectBoxBox.prototype.getSupportPoint=function(box, axis) {
126 		var orientationCol = box.get_currentState().getOrientationCols();
127 		var _as = Vector3DUtil.dotProduct(axis,orientationCol[0]);
128 		var _au = Vector3DUtil.dotProduct(axis,orientationCol[1]);
129 		var _ad = Vector3DUtil.dotProduct(axis,orientationCol[2]);
130 						
131 		var p = box.get_currentState().position.slice(0);
132   
133 		var sideLengths=box.get_sideLengths();
134 		
135 		if (_as < -JNumber3D.NUM_TINY) {
136 			p = Vector3DUtil.add(p,JNumber3D.getScaleVector(orientationCol[0], 0.5 * sideLengths[0]));
137 		}else if (_as >= JNumber3D.NUM_TINY) {
138 			p = Vector3DUtil.subtract(p,JNumber3D.getScaleVector(orientationCol[0], 0.5 * sideLengths[0]));
139 		}
140   
141 		if (_au < -JNumber3D.NUM_TINY) {
142 			p = Vector3DUtil.add(p,JNumber3D.getScaleVector(orientationCol[1], 0.5 * sideLengths[1]));
143 		}else if (_au > JNumber3D.NUM_TINY) {
144 			p = Vector3DUtil.subtract(p,JNumber3D.getScaleVector(orientationCol[1], 0.5 * sideLengths[1]));
145 		}
146   
147 		if (_ad < -JNumber3D.NUM_TINY) {
148 			p = Vector3DUtil.add(p,JNumber3D.getScaleVector(orientationCol[2], 0.5 * sideLengths[2]));
149 		}else if (_ad > JNumber3D.NUM_TINY) {
150 			p = Vector3DUtil.subtract(p,JNumber3D.getScaleVector(orientationCol[2], 0.5 * sideLengths[2]));
151 		}
152 		return p;
153 	};

155 	/**
156 	 * @function getAABox2EdgeIntersectionPoints
157 	 * @param {array} contactPoint a 3D vector
158 	 * @param {array} origBoxSides a 3D vector
159 	 * @param {PhysicsState} origBoxState
160 	 * @param {array} edgePt0 a 3D vector
161 	 * @param {array} edgePt1 a 3D vector
162 	 * @type {number}
163 	 **/
164 	CollDetectBoxBox.prototype.getAABox2EdgeIntersectionPoints=function(contactPoint, origBoxSides, origBoxState, edgePt0, edgePt1){
165 		var jDir;
166 		var kDir;
167 		var dist0;
168 		var dist1;
169 		var frac;
170 		var num=0;
171 		var pt;
172 		var edgeDir = Vector3DUtil.subtract(edgePt1, edgePt0);
173 		Vector3DUtil.normalize(edgeDir);
174 		var ptArr=[];
175 		var faceOffsets=[];
176 		var edgePt0Arr = edgePt0;
177 		var edgePt1Arr = edgePt1;
178 		var edgeDirArr = edgeDir;
179 		var sidesArr = JNumber3D.getScaleVector(origBoxSides, 0.5);
180 		for (var iDir= 2; iDir >= 0; iDir--) {
181 			if (Math.abs(edgeDirArr[iDir]) < 0.1) {
182 				continue;
183 			}
184 			jDir = (iDir + 1) % 3;
185 			kDir = (iDir + 2) % 3;
186 			faceOffsets = [ -sidesArr[iDir], sidesArr[iDir]];
187 			for (var iFace= 1; iFace >= 0; iFace-- ) {
188 				dist0 = edgePt0Arr[iDir] - faceOffsets[iFace];
189 				dist1 = edgePt1Arr[iDir] - faceOffsets[iFace];
190 				frac = -1;
191 				if (dist0 * dist1 < -JNumber3D.NUM_TINY) {
192 					frac = -dist0 / (dist1 - dist0);
193 				}else if (Math.abs(dist0) < JNumber3D.NUM_TINY) {
194 					frac = 0;
195 				}else if (Math.abs(dist1) < JNumber3D.NUM_TINY) {
196 					frac = 1;
197 				}
198 				if (frac >= 0) {
199 					pt = Vector3DUtil.add(JNumber3D.getScaleVector(edgePt0, 1 - frac),JNumber3D.getScaleVector(edgePt1, frac));
200 					ptArr = pt;
201 					if ((ptArr[jDir] > -sidesArr[jDir] - JNumber3D.NUM_TINY) && (ptArr[jDir] < sidesArr[jDir] + JNumber3D.NUM_TINY) && (ptArr[kDir] > -sidesArr[kDir] - JNumber3D.NUM_TINY) && (ptArr[kDir] < sidesArr[kDir] + JNumber3D.NUM_TINY) ) {
202 						pt=pt.splice(0);
203 						JMatrix3D.multiplyVector(origBoxState.get_orientation(),pt);
204 						pt = Vector3DUtil.add(pt,origBoxState.position);
205 						this.addPoint(contactPoint, pt, combinationDist);
206 						if (++num == 2) {
207 							return num;
208 						}
209 					}
210 				}
211 			}
212 		}
213 		return num;
214 	};

216 	/**
217 	 * @function getBox2BoxEdgesIntersectionPoints
218 	 * @param {array} contactPoint a 3D vector
219 	 * @param {JBox} box0
220 	 * @param {JBox} box1
221 	 * @param {PhysicsState} newState
222 	 * @type {number}
223 	 **/
224 	CollDetectBoxBox.prototype.getBox2BoxEdgesIntersectionPoints=function(contactPoint, box0, box1, newState){
225 		var num = 0;
226 		var seg;
227 		var box0State = (newState) ? box0.get_currentState() : box0.get_oldState();
228 		var box1State= (newState) ? box1.get_currentState() : box1.get_oldState();
229 		var boxPts = box1.getCornerPointsInBoxSpace(box1State, box0State);
230 		
231 		var boxEdges = box1.get_edges();
232 		var edgePt0;
233 		var edgePt1;
234 		for(var i=0;i<boxEdges.length;i++){
235 		var boxEdge=boxEdges[i];
236 			edgePt0 = boxPts[boxEdge.ind0];
237 			edgePt1 = boxPts[boxEdge.ind1];
238 			num += this.getAABox2EdgeIntersectionPoints(contactPoint, box0.get_sideLengths(), box0State, edgePt0, edgePt1);
239 			if (num >= 8) {
240 				return num;
241 			}
242 		}
243 		return num;
244 	};

257 		return Vector3DUtil.get_length(contactPoint);

260 	/**
261 	 * @function collDetect detects a collision and updates the info parameter
262 	 * @param {CollDetectInfo} info
263 	 * @param {array} collArray
264 	 * @param {PhysicsState} newState
265 	 * @type {void}
266 	 **/
267 	CollDetectBoxBox.prototype.collDetect=function(info, collArr){
268 		var box0 = info.body0;
269 		var box1 = info.body1;
270 
271 		if (!box0.hitTestObject3D(box1)) return;
272 
273 		if (JConfig.aabbDetection && !box0.get_boundingBox().overlapTest(box1.get_boundingBox())) return;
274 
275 		var numTiny= JNumber3D.NUM_TINY;
276 		var numHuge= JNumber3D.NUM_HUGE;
277 
278 		var dirs0Arr = box0.get_currentState().getOrientationCols();
279 		var dirs1Arr = box1.get_currentState().getOrientationCols();
280 
281 		// the 15 potential separating axes
282 		var axes = [dirs0Arr[0], dirs0Arr[1], dirs0Arr[2],
283 								dirs1Arr[0], dirs1Arr[1], dirs1Arr[2],
284 								Vector3DUtil.crossProduct(dirs0Arr[0],dirs1Arr[0]),
285 								Vector3DUtil.crossProduct(dirs0Arr[1],dirs1Arr[0]),
286 								Vector3DUtil.crossProduct(dirs0Arr[2],dirs1Arr[0]),
287 								Vector3DUtil.crossProduct(dirs0Arr[0],dirs1Arr[1]),
288 								Vector3DUtil.crossProduct(dirs0Arr[1],dirs1Arr[1]),
289 								Vector3DUtil.crossProduct(dirs0Arr[2],dirs1Arr[1]),
290 								Vector3DUtil.crossProduct(dirs0Arr[0],dirs1Arr[2]),
291 								Vector3DUtil.crossProduct(dirs0Arr[1],dirs1Arr[2]),
292 								Vector3DUtil.crossProduct(dirs0Arr[2],dirs1Arr[2])];
293 
294 		var l2;
295 		// the overlap depths along each axis
296 		var overlapDepths = [];
297 		var i= 0;
298 		var axesLength = axes.length;
299 
300 		// see if the boxes are separate along any axis, and if not keep a 
301 		// record of the depths along each axis
302 		for (i = 0; i < axesLength; i++){
303 			var _overlapDepth = overlapDepths[i] = new SpanData();
304 			_overlapDepth.depth = numHuge;
305 
306 			l2 = Vector3DUtil.get_lengthSquared(axes[i]);
307 			if (l2 < numTiny) continue;
308 								
309 			var ax = axes[i].slice(0);
310 			Vector3DUtil.normalize(ax);
311 			if (this.disjoint(overlapDepths[i], ax, box0, box1)) return;
312 		}
313 
314 		// The box overlap, find the separation depth closest to 0.
315 		var minDepth = numHuge;
316 		var minAxis = -1;
317 		axesLength = axes.length;
318 		for (i = 0; i < axesLength; i++){
319 			l2 = Vector3DUtil.get_lengthSquared(axes[i]);
320 			if (l2 < numTiny) continue;
321 
322 			// If this axis is the minimum, select it
323 			if (overlapDepths[i].depth < minDepth){
324 				minDepth = overlapDepths[i].depth;
325 				minAxis =i;
326 			}
327 		}
328 						
329 		if (minAxis == -1) return;
330 						
331 		// Make sure the axis is facing towards the box0. if not, invert it
332 		var N= axes[minAxis].splice(0);
333 		if (Vector3DUtil.dotProduct(Vector3DUtil.subtract(box1.get_currentState().position,box0.get_currentState().position),N) > 0) 
334 			N = JNumber3D.getScaleVector(N, -1);
335 						
336 		var contactPointsFromOld = true;
337 		var contactPoints = [];
338 		var box0lengths=box0.get_sideLengths();
339 		var box1lengths=box1.get_sideLengths();
340 		combinationDist = 0.05 * Math.min(Math.min(box0lengths[0], box0lengths[1], box0lengths[2]), Math.min(box1lengths[0], box1lengths[1], box1lengths[2]));
341 		combinationDist += (JConfig.collToll * 3.464);
342 		combinationDist *= combinationDist;
343 
344 		if (minDepth > -JNumber3D.NUM_TINY)
345 			this.getBoxBoxIntersectionPoints(contactPoints, box0, box1, false);
346 						
347 		if (contactPoints.length == 0){
348 			contactPointsFromOld = false;
349 			this.getBoxBoxIntersectionPoints(contactPoints, box0, box1, true);
350 		}
351 						
352 		var bodyDelta = Vector3DUtil.subtract(Vector3DUtil.subtract(box0.get_currentState().position,box0.get_oldState().position),Vector3DUtil.subtract(box1.get_currentState().position,box1.get_oldState().position));
353 		var bodyDeltaLen = Vector3DUtil.dotProduct(bodyDelta,N);
354 		var oldDepth = minDepth + bodyDeltaLen;
355 						
356 		var SATPoint = [];
357 		switch(minAxis){
358 			//-----------------------------------------------------------------
359 			// Box0 face, Box1 Corner collision
360 			//-----------------------------------------------------------------
361 		case 0:
362 		case 1:
363 		case 2:
364 			//-----------------------------------------------------------------
365 			// Get the lowest point on the box1 along box1 normal
366 			//-----------------------------------------------------------------
367 			SATPoint = this.getSupportPoint(box1, JNumber3D.getScaleVector(N, -1));
368 			break;
369 		//-----------------------------------------------------------------
370 		// We have a Box2 corner/Box1 face collision
371 		//-----------------------------------------------------------------
372 		case 3:
373 		case 4:
374 		case 5:
375 			//-----------------------------------------------------------------
376 			// Find with vertex on the triangle collided
377 			//-----------------------------------------------------------------
378 			SATPoint = this.getSupportPoint(box0, N);
379 			break;
380 		//-----------------------------------------------------------------
381 		// We have an edge/edge colliiosn
382 		//-----------------------------------------------------------------
383 		case 6:
384 		case 7:
385 		case 8:
386 		case 9:
387 		case 10:
388 		case 11:
389 		case 12:
390 		case 13:
391 		case 14:
392 			//-----------------------------------------------------------------
393 			// Retrieve which edges collided.
394 			//-----------------------------------------------------------------
395 			i = minAxis - 6;
396 			var ia = (i / 3)|0;
397 			var ib= i - ia * 3;
398 			//-----------------------------------------------------------------
399 			// find two P0, P1 point on both edges. 
400 			//-----------------------------------------------------------------
401 			var P0 = this.getSupportPoint(box0, N);
402 			var P1 = this.getSupportPoint(box1, JNumber3D.getScaleVector(N, -1));
403 	  
404 			//-----------------------------------------------------------------
405 			// Find the edge intersection. 
406 			//-----------------------------------------------------------------
407 	 
408 			//-----------------------------------------------------------------
409 			// plane along N and F, and passing through PB
410 			//-----------------------------------------------------------------
411 			var planeNormal = Vector3DUtil.crossProduct(N,dirs1Arr[ib]);
412 			var planeD = Vector3DUtil.dotProduct(planeNormal,P1);
413 	  
414 			//-----------------------------------------------------------------
415 			// find the intersection t, where Pintersection = P0 + t*box edge dir
416 			//-----------------------------------------------------------------
417 			var div = Vector3DUtil.dotProduct(dirs0Arr[ia],planeNormal);
418 	  
419 			//-----------------------------------------------------------------
420 			// plane and ray colinear, skip the intersection.
421 			//-----------------------------------------------------------------
422 			if (Math.abs(div) < JNumber3D.NUM_TINY) return;
423 	  
424 			var t = (planeD - Vector3DUtil.dotProduct(P0,planeNormal)) / div;
425 	  
426 			//-----------------------------------------------------------------
427 			// point on edge of box0
428 			//-----------------------------------------------------------------
429 			P0 = Vector3DUtil.add(P0,JNumber3D.getScaleVector(dirs0Arr[ia], t));
430 			SATPoint =Vector3DUtil.add(P0,JNumber3D.getScaleVector(N, 0.5 * minDepth));
431 			break;
432 		}
433 
434 		var collPts;
435 		if (contactPoints.length > 0){
436 			collPts = [];
437 
438 			var minDist = JNumber3D.NUM_HUGE;
439 			var maxDist = -JNumber3D.NUM_HUGE;
440 			var dist;
441 			var depth;
442 			var depthScale;
443 			var cpInfo;
444 			var contactPoint;
445 
446 			for(var j=0;j<contactPoints.length;j++){
447 				contactPoint=contactPoints[j];
448 
449 				dist = Vector3DUtil.get_length(Vector3DUtil.subtract(contactPoint,SATPoint));
450 						
451 				if (dist < minDist) minDist = dist;
452 
453 				if (dist > maxDist) maxDist = dist;
454 			}
455 
456 			if (maxDist < minDist + JNumber3D.NUM_TINY) maxDist = minDist + JNumber3D.NUM_TINY;
457 
458 			i = 0;
459 							  
460 			for(var j=0;j<contactPoints.length;j++){
461 				contactPoint=contactPoints[j];
462 				dist = Vector3DUtil.get_length(Vector3DUtil.subtract(contactPoint,SATPoint));
463 				depthScale = (dist - minDist) / (maxDist - minDist);
464 				depth = (1 - depthScale) * oldDepth;
465 				cpInfo = new CollPointInfo();
466 						
467 				if (contactPointsFromOld) {
468 					cpInfo.r0 = Vector3DUtil.subtract(contactPoint,box0.get_oldState().position);
469 					cpInfo.r1 = Vector3DUtil.subtract(contactPoint,box1.get_oldState().position);
470 				} else {
471 					cpInfo.r0 = Vector3DUtil.subtract(contactPoint,box0.get_currentState().position);
472 					cpInfo.r1 = Vector3DUtil.subtract(contactPoint,box1.get_currentState().position);
473 				}
474 						
475 				cpInfo.initialPenetration = depth;
476 				collPts[i++] = cpInfo;
477 			}
478 		}else{
479 			cpInfo = new CollPointInfo();
480 			cpInfo.r0 = Vector3DUtil.subtract(SATPoint,box0.get_currentState().position);
481 			cpInfo.r1 = Vector3DUtil.subtract(SATPoint,box1.get_currentState().position);
482 			cpInfo.initialPenetration = oldDepth;
483 
484 			collPts = [];
485 			collPts[0] = cpInfo;
486 		}
487 		var collInfo = new CollisionInfo();
488 		collInfo.objInfo = info;
489 		collInfo.dirToBody = N;
490 		collInfo.pointInfo = collPts;
491 
492 		var mat = new MaterialProperties();
493 		mat.set_restitution(Math.sqrt(box0.get_material().get_restitution() * box1.get_material().get_restitution()));
494 		mat.set_friction(Math.sqrt(box0.get_material().get_friction() * box1.get_material().get_friction()));
495 		collInfo.mat = mat;
496 		collArr.push(collInfo);
497 
498 		info.body0.collisions.push(collInfo);
499 		info.body1.collisions.push(collInfo);
500 	};
501 
502 	jigLib.CollDetectBoxBox=CollDetectBoxBox;
