 21 (function(jigLib){
 22 	var Vector3DUtil=jigLib.Vector3DUtil;
 23 	var JMatrix3D=jigLib.JMatrix3D;
 24 	var JNumber3D=jigLib.JNumber3D;
 25 	var JConstraint=jigLib.JConstraint;
 26 	var JConfig=jigLib.JConfig;
 27 	var JCapsule=jigLib.JCapsule;
 28 	var JSegment=jigLib.JSegment;
 29 	var JBox=jigLib.JBox;
 30 	var MaterialProperties=jigLib.MaterialProperties;
 31 	var RigidBody=jigLib.RigidBody;
 32 	var CollPointInfo=jigLib.CollPointInfo;
 33 	var CollisionInfo=jigLib.CollisionInfo;
 34 	
 35 	/**
 36 	 * @author Muzer(muzerly@gmail.com)
 37 	 * 
 38 	 * @name CollDetectCapsuleBox
 39 	 * @class CollDetectCapsuleBox handles collisions between capsules and boxes
 40 	 * @extends CollDetectFunctor
 41 	 * @requires CollDetectInfo
 42 	 * @requires CollPointInfo
 43 	 * @requires Vector3DUtil
 44 	 * @requires JNumber3D
 45 	 * @requires JSegment
 46 	 * @requires MaterialProperties
 47 	 * @constructor
 48 	 **/
 49 	var CollDetectCapsuleBox=function(){
 50 		this.name = "CapsuleBox";
 51 		this.type0 = "CAPSULE";
 52 		this.type1 = "BOX";
 53 	};
 54 	jigLib.extend(CollDetectCapsuleBox,jigLib.CollDetectFunctor);
 55 
 56 	/**
 57 	 * @function collDetect detects a collision and updates the info parameter
 58 	 * @param {CollDetectInfo} info
 59 	 * @param {array} collArray
 60 	 * @type void
 61 	 **/
 62 	CollDetectCapsuleBox.prototype.collDetect=function(info, collArr){
 63 		var tempBody;
 64 		if (info.body0.get_type() == "BOX"){
 65 			tempBody = info.body0;
 66 			info.body0 = info.body1;
 67 			info.body1 = tempBody;
 68 		}
 69 
 70 		var capsule = info.body0;
 71 		var box = info.body1;
 72 
 73 		if (!capsule.hitTestObject3D(box)){
 74 			return;
 75 		}
 76 		if (JConfig.aabbDetection && !capsule.get_boundingBox().overlapTest(box.get_boundingBox())) {
 77 			return;
 78 		}
 79 
 80 		var collPts = [];
 81 		var cpInfo;
 82 
 83 		var averageNormal = [0,0,0,0];
 84 		var oldSeg = new JSegment(capsule.getEndPos(capsule.get_oldState()), JNumber3D.getScaleVector(capsule.get_oldState().getOrientationCols()[1], -Vector3DUtil.get_length(capsule)));
 85 		var newSeg = new JSegment(capsule.getEndPos(capsule.get_currentState()), JNumber3D.getScaleVector(capsule.get_currentState().getOrientationCols()[1], -Vector3DUtil.get_length(capsule)));
 86 		var radius = capsule.get_radius();
 87 
 88 		var oldObj = {};
 89 		var oldDistSq= oldSeg.segmentBoxDistanceSq(oldObj, box, box.get_oldState());
 90 		var newObj = {};
 91 		var newDistSq = newSeg.segmentBoxDistanceSq(newObj, box, box.get_currentState());
 92 		var arr = box.get_oldState().getOrientationCols();
 93 
 94 		if (Math.min(oldDistSq, newDistSq) < Math.pow(radius + JConfig.collToll, 2)){
 95 			var segPos = oldSeg.getPoint(Number(oldObj.pfLParam));
 96 			var boxPos = box.get_oldState().position.slice(0);
 97 			boxPos = Vector3DUtil.add(boxPos, JNumber3D.getScaleVector(arr[0], oldObj.pfLParam0));
 98 			boxPos = Vector3DUtil.add(boxPos, JNumber3D.getScaleVector(arr[1], oldObj.pfLParam1));
 99 			boxPos = Vector3DUtil.add(boxPos, JNumber3D.getScaleVector(arr[2], oldObj.pfLParam2));
100 
101 			var dist = Math.sqrt(oldDistSq);
102 			var depth = radius - dist;
103 
104 			var dir;
105 			if (dist > JNumber3D.NUM_TINY){
106 				dir = Vector3DUtil.subtract(segPos, boxPos);
107 				Vector3DUtil.normalize(dir);
108 			}else if (Vector3DUtil.get_length(Vector3DUtil.subtract(segPos, box.get_oldState().position)) > JNumber3D.NUM_TINY){
109 				dir = Vector3DUtil.subtract(segPos, box.get_oldState().position);
110 				Vector3DUtil.normalize(dir);
111 			}else{
112 				dir = Vector3DUtil.Y_AXIS;
113 				JMatrix3D.multiplyVector(JMatrix3D.getRotationMatrix(0, 0, 1, 360 * Math.random()), dir);
114 			}
115 			averageNormal = Vector3DUtil.add(averageNormal, dir);
116 
117 			cpInfo = new CollPointInfo();
118 			cpInfo.r0 = Vector3DUtil.subtract(boxPos, capsule.get_oldState().position);
119 			cpInfo.r1 = Vector3DUtil.subtract(boxPos, box.get_oldState().position);
120 			cpInfo.initialPenetration = depth;
121 			collPts.push(cpInfo);
122 		}
123 
124 
125 		oldSeg = new JSegment(capsule.getBottomPos(capsule.get_oldState()), JNumber3D.getScaleVector(capsule.get_oldState().getOrientationCols()[1], Vector3DUtil.get_length(capsule)));
126 		newSeg = new JSegment(capsule.getBottomPos(capsule.get_currentState()), JNumber3D.getScaleVector(capsule.get_currentState().getOrientationCols()[1], Vector3DUtil.get_length(capsule)));
127 
128 		oldObj = {};
129 		oldDistSq = oldSeg.segmentBoxDistanceSq(oldObj, box, box.get_oldState());
130 		newObj = {};
131 		newDistSq = newSeg.segmentBoxDistanceSq(newObj, box, box.get_currentState());
132 
133 		if (Math.min(oldDistSq, newDistSq) < Math.pow(radius + JConfig.collToll, 2)){
134 			segPos = oldSeg.getPoint(Number(oldObj.pfLParam));
135 			boxPos = box.get_oldState().position.slice(0);
136 			boxPos = Vector3DUtil.add(boxPos, JNumber3D.getScaleVector(arr[0], oldObj.pfLParam0));
137 			boxPos = Vector3DUtil.add(boxPos, JNumber3D.getScaleVector(arr[1], oldObj.pfLParam1));
138 			boxPos = Vector3DUtil.add(boxPos, JNumber3D.getScaleVector(arr[2], oldObj.pfLParam2));
139 
140 			dist = Math.sqrt(oldDistSq);
141 			depth = radius - dist;
142 
143 			if (dist > JNumber3D.NUM_TINY){
144 				dir = Vector3DUtil.subtract(segPos, boxPos);
145 				Vector3DUtil.normalize(dir);
146 			}else if (Vector3DUtil.get_length(Vector3DUtil.subtract(segPos, box.get_oldState().position)) > JNumber3D.NUM_TINY){
147 				dir = Vector3DUtil.subtract(segPos, box.get_oldState().position);
148 				Vector3DUtil.normalize(dir);
149 			}else{
150 				dir = Vector3DUtil.Y_AXIS;
151 				JMatrix3D.multiplyVector(JMatrix3D.getRotationMatrix(0, 0, 1, 360 * Math.random()), dir);
152 			}
153 			averageNormal = Vector3DUtil.add(averageNormal, dir);
154 
155 			cpInfo = new CollPointInfo();
156 			cpInfo.r0 = Vector3DUtil.subtract(boxPos, capsule.get_oldState().position);
157 			cpInfo.r1 = Vector3DUtil.subtract(boxPos, box.get_oldState().position);
158 			cpInfo.initialPenetration = depth;
159 			collPts.push(cpInfo);
160 		}
161 
162 		if (collPts.length > 0){
163 			averageNormal.normalize();
164 			var collInfo = new CollisionInfo();
165 			collInfo.objInfo = info;
166 			collInfo.dirToBody = averageNormal;
167 			collInfo.pointInfo = collPts;
168 
169 			var mat = new MaterialProperties();
170 			mat.set_restitution(Math.sqrt(capsule.get_material().get_restitution() * box.get_material().get_restitution()));
171 			mat.set_friction(Math.sqrt(capsule.get_material().get_friction() * box.get_material().get_friction()));
172 			collInfo.mat = mat;
173 			collArr.push(collInfo);
174 
175 			info.body0.collisions.push(collInfo);
176 			info.body1.collisions.push(collInfo);
177 		}
178 	};
179 	
180 	jigLib.CollDetectCapsuleBox=CollDetectCapsuleBox;
181 	
182 })(jigLib);
