 21 (function(jigLib){
 22 	var Vector3DUtil=jigLib.Vector3DUtil;
 23 	var JNumber3D=jigLib.JNumber3D;
 24 	var JConstraint=jigLib.JConstraint;
 25 	var JConfig=jigLib.JConfig;
 26 	var JCapsule=jigLib.JCapsule;
 27 	var JTerrain=jigLib.JPlane;
 28 	var MaterialProperties=jigLib.MaterialProperties;
 29 	var RigidBody=jigLib.RigidBody;
 30 	var CollPointInfo=jigLib.CollPointInfo;
 31 	var CollisionInfo=jigLib.CollisionInfo;
 32 	
 33 	/**
 34 	 * @author Muzer(muzerly@gmail.com)
 35 	 * 
 36 	 * @name CollDetectCapsulePlane
 37 	 * @class CollDetectCapsulePlane handles collisions between capsules and planes
 38 	 * @extends CollDetectFunctor
 39 	 * @requires CollDetectInfo
 40 	 * @requires Vector3DUtil
 41 	 * @requires JNumber3D
 42 	 * @requires MaterialProperties
 43 	 * @constructor
 44 	 **/
 45 	var CollDetectCapsulePlane=function(){
 46 		this.name = "CapsulePlane";
 47 		this.type0 = "CAPSULE";
 48 		this.type1 = "PLANE";
 49 	};
 50 	jigLib.extend(CollDetectCapsulePlane,jigLib.CollDetectFunctor);
 51 	
 52 	/**
 53 	 * @function collDetect detects a collision and updates the info parameter
 54 	 * @param {CollDetectInfo} info
 55 	 * @param {array} collArray
 56 	 * @type void
 57 	 **/
 58 	CollDetectCapsulePlane.prototype.collDetect=function(info, collArr){
 59 		var tempBody;
 60 		if (info.body0.get_type() == "PLANE"){
 61 			tempBody = info.body0;
 62 			info.body0 = info.body1;
 63 			info.body1 = tempBody;
 64 		}
 65 
 66 		var capsule = info.body0;
 67 		var plane = info.body1;
 68 
 69 		var collPts = [];
 70 		var cpInfo;
 71 
 72 		var oldPos = capsule.getBottomPos(capsule.get_oldState());
 73 		var oldDist = plane.pointPlaneDistance(oldPos);
 74 		var newPos = capsule.getBottomPos(capsule.get_currentState());
 75 		var newDist = plane.pointPlaneDistance(newPos);
 76 
 77 		if (Math.min(oldDist, newDist) < capsule.get_radius() + JConfig.collToll){
 78 			var oldDepth= capsule.get_radius() - oldDist;
 79 			var worldPos= Vector3DUtil.subtract(oldPos, JNumber3D.getScaleVector(plane.get_normal(), capsule.get_radius()));
 80 
 81 			cpInfo = new CollPointInfo();
 82 			cpInfo.r0 = Vector3DUtil.subtract(worldPos, capsule.get_oldState().position);
 83 			cpInfo.r1 = Vector3DUtil.subtract(worldPos, plane.get_oldState().position);
 84 			cpInfo.initialPenetration = oldDepth;
 85 			collPts.push(cpInfo);
 86 		}
 87 
 88 		oldPos = capsule.getEndPos(capsule.get_oldState());
 89 		newPos = capsule.getEndPos(capsule.get_currentState());
 90 		oldDist = plane.pointPlaneDistance(oldPos);
 91 		newDist = plane.pointPlaneDistance(newPos);
 92 		if (Math.min(oldDist, newDist) < capsule.get_radius() + JConfig.collToll){
 93 			oldDepth = capsule.get_radius() - oldDist;
 94 			worldPos = Vector3DUtil.subtract(oldPos, JNumber3D.getScaleVector(plane.get_normal(), capsule.get_radius()));
 95 
 96 			cpInfo = new CollPointInfo();
 97 			cpInfo.r0 = Vector3DUtil.subtract(worldPos, capsule.get_oldState().position);
 98 			cpInfo.r1 = Vector3DUtil.subtract(worldPos, plane.get_oldState().position);
 99 			cpInfo.initialPenetration = oldDepth;
100 			collPts.push(cpInfo);
101 		}
102 
103 		if (collPts.length > 0){
104 			var collInfo= new CollisionInfo();
105 			collInfo.objInfo = info;
106 			collInfo.dirToBody = plane.get_normal();
107 			collInfo.pointInfo = collPts;
108 
109 			var mat = new MaterialProperties();
110 			mat.set_restitution(Math.sqrt(capsule.get_material().get_restitution() * plane.get_material().get_restitution()));
111 			mat.set_friction(Math.sqrt(capsule.get_material().get_friction() * plane.get_material().get_friction()));
112 			collInfo.mat = mat;
113 			collArr.push(collInfo);
114 
115 			info.body0.collisions.push(collInfo);
116 			info.body1.collisions.push(collInfo);
117 		}
118 	};
119 	
120 	jigLib.CollDetectCapsulePlane=CollDetectCapsulePlane;
121 	
122 })(jigLib);
