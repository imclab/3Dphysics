1 (function(jigLib){
  2 	var Vector3DUtil=jigLib.Vector3DUtil;

  5 	var JSphere=jigLib.JSphere;
  6 	var JTriangle=jigLib.JTriangle;
  7 	var JTriangleMesh=jigLib.JTriangleMesh;

 14         var CollDetectSphereMesh=function() {
 15 		this.name = "SphereMesh";
 16 		this.type0 = "SPHERE";
 17 		this.type1 = "TRIANGLEMESH";
 18 	}	
 19 	
 20 	jigLib.extend(CollDetectSphereMesh,jigLib.CollDetectFunctor);
 21

 23 	CollDetectSphereMesh.prototype.collDetectSphereStaticMeshOverlap=function(sphere, mesh, info, collTolerance, collArr){
 24 		var body0Pos = info.body0.get_oldState().position;
 25 		var body1Pos = info.body1.get_oldState().position;
 26 
 27 		var sphereTolR = collTolerance + sphere.get_radius();
 28 		var sphereTolR2 = sphereTolR * sphereTolR;
 29                         
 30 		var collNormal = [0,0,0];
 31 		var collPts = []
 32                         
 33 		var potentialTriangles = [];
 34 		var numTriangles = mesh.get_octree().getTrianglesIntersectingtAABox(potentialTriangles, sphere.get_boundingBox());
 35                         
 36 		var newD2,distToCentre,oldD2,dist,depth,tiny=JNumber3D.NUM_TINY;
 37 		var meshTriangle;
 38 		var vertexIndices;
 39 		var arr;
 40 		var triangle;
 41 		for (var iTriangle = 0 ; iTriangle < numTriangles ; ++iTriangle) {
 42 			meshTriangle = mesh.get_octree().getTriangle(potentialTriangles[iTriangle]);
 43 			distToCentre = meshTriangle.get_plane().pointPlaneDistance(sphere.get_currentState().position);
 44 			//alert(distToCentre);
 45 			if (distToCentre <= 0) continue;
 46 			if (distToCentre >= sphereTolR) continue;
 47                                 
 48 			vertexIndices = meshTriangle.get_vertexIndices();
 49 			triangle = new JTriangle(mesh.get_octree().getVertex(vertexIndices[0]), mesh.get_octree().getVertex(vertexIndices[1]), mesh.get_octree().getVertex(vertexIndices[2]));
 50 			arr = [];
 51 			newD2 = triangle.pointTriangleDistanceSq(arr, sphere.get_currentState().position);
 52                                 
 53 			if (newD2 < sphereTolR2) {
 54 				// have overlap - but actually report the old intersection
 55 				oldD2 = triangle.pointTriangleDistanceSq(arr, sphere.get_oldState().position);
 56 				dist = Math.sqrt(oldD2);
 57 				depth = sphere.get_radius() - dist;
 58 				var collisionN = (dist > tiny) ? (Vector3DUtil.subtract(sphere.get_oldState().position,triangle.getPoint(arr[0], arr[1]))) : triangle.normal.slice(0);
 59 				Vector3DUtil.normalize(collisionN);
 60 				// since impulse get applied at the old position
 61 				var pt = Vector3DUtil.subtract(sphere.get_oldState().position,JNumber3D.getScaleVector(collisionN, sphere.get_radius()));
 62                                         
 63 				var cpInfo = new CollPointInfo();
 64 				cpInfo.r0 = Vector3DUtil.subtract(pt,body0Pos);
 65 				cpInfo.r1 = Vector3DUtil.subtract(pt,body1Pos);
 66 				cpInfo.initialPenetration = depth;
 67 				collPts.push(cpInfo);
 68 				collNormal = Vector3DUtil.add(collNormal,collisionN);
 69 				Vector3DUtil.normalize(collNormal);
 70 			}
 71 		}
 72                         
 73 		var collInfo = new jigLib.CollisionInfo();
 74 		collInfo.objInfo = info;
 75 		collInfo.dirToBody = collNormal;
 76 		collInfo.pointInfo = collPts;
 77                         
 78 		var mat = new MaterialProperties();
 79 		mat.set_restitution(0.5*(sphere.get_material().get_restitution() + mesh.get_material().get_restitution()));
 80 		mat.set_friction(0.5*(sphere.get_material().get_friction() + mesh.get_material().get_friction()));
 81 		collInfo.mat = mat;
 82 		collArr.push(collInfo);
 83 		info.body0.collisions.push(collInfo);
 84 		info.body1.collisions.push(collInfo);
 85 	};
 86

 88 	CollDetectSphereMesh.prototype.collDetect=function(info, collArr){
 89 		var tempBody;
 90 		if (info.body0._type == "TRIANGLEMESH"){
 91 			tempBody = info.body0;
 92 			info.body0 = info.body1;
 93 			info.body1 = tempBody;
 94 		}
 95                         
 96 		var sphere = info.body0;
 97 		var mesh = info.body1;
 98                         
 99 		this.collDetectSphereStaticMeshOverlap(sphere, mesh, info, JConfig.collToll, collArr);
100 	}

102 	jigLib.CollDetectSphereMesh=CollDetectSphereMesh;
