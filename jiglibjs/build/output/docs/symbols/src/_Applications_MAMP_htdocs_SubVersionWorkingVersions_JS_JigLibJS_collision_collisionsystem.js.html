106 	 * @function addCollisionBody adds a rigid body to the colBody collection

108 	 * @type void
109 	 **/
110 	CollisionSystem.prototype.addCollisionBody=function(body){
111 		if (!this.findBody(body))
112 			this.collBody.push(body);
113 	};

116 	 * @function removeCollisionBody removes a rigid body from the colBody collection
117 	 * @param {RigidBody} body
118 	 * @type void
119 	 **/
120 	CollisionSystem.prototype.removeCollisionBody=function(body){
121 		if (this.findBody(body))
122 			this.collBody.splice(this.collBody.indexOf(body), 1);
123 	};

133 	/**
134 	 * @function detectCollisions detects collisions between the body and all the registered collision bodies
135 	 * @param {RigidBody} body
136 	 * @param {array} collArr
137 	 * @type void
138 	 **/
139 	CollisionSystem.prototype.detectCollisions=function(body, collArr){
140 		if (!body.isActive)
141 			return;
142 
143 		var info;
144 		var fu;
145 
146 		for(var i=0, cbl=this.collBody.length; i<cbl; i++){
147 			var _collBody=this.collBody[i];
148 			if (body != _collBody && this.checkCollidables(body, _collBody) && this.detectionFunctors[body.get_type()][_collBody.get_type()] != undefined){
149 				info = new CollDetectInfo();
150 				info.body0 = body;
151 				info.body1 = _collBody;
152 				fu = detectionFunctors[info.body0.get_type()][info.body1.get_type()];
153 				fu.collDetect(info, collArr);
154 			}
155 		}
156 	};
157 	
158 	/**
159 	 * @function detectAllCollisions detects collisions between all bodies
160 	 * @param {array} bodies
161 	 * @param {array} collArr
162 	 * @type void
163 	 **/
164 	CollisionSystem.prototype.detectAllCollisions=function(bodies, collArr){
165 		var info;
166 		var fu;
167 		var bodyID;
168 		var bodyType;
169 						
170 		for(var i=0, bl=bodies.length; i<bl; i++){
171 			var _body=bodies[i];
172 			bodyID = _body.id;
173 			bodyType = _body.get_type();
174 
175 			for(var j=0;j<this.collBody.length;j++){
176 				var _collBody=this.collBody[j];
177 				if (_body == _collBody){
178 					continue;
179 				}
180 
181 				if (_collBody.isActive && bodyID > _collBody.id){
182 					continue;
183 				}
184 
185 				if (this.checkCollidables(_body, _collBody) && this.detectionFunctors[bodyType][_collBody.get_type()] != undefined){
186 					info = new CollDetectInfo();
187 					info.body0 = _body;
188 					info.body1 = _collBody;
189 					fu = detectionFunctors[info.body0.get_type()][info.body1.get_type()];
190 					fu.collDetect(info, collArr);
191 				}
192 			}
193 		}
194 	};

196 	/**
197 	 * @function segmentIntersect
198 	 * @param {object} out
199 	 * @param {JSegment} seg
200 	 * @param {RigidBody} ownerBody
201 	 * @type boolean
202 	 **/
203 	CollisionSystem.prototype.segmentIntersect=function(out, seg, ownerBody){
204 		out.fracOut = JNumber3D.NUM_HUGE;
205 		out.posOut = [0,0,0,0];
206 		out.normalOut = [0,0,0,0];
207 
208 		var obj= {};
209 
210 		for(var i=0, cbl=this.collBody.length; i<cbl; i++){
211 			var _collBody=this.collBody[i];
212 			if (_collBody != ownerBody && this.segmentBounding(seg, _collBody)){
213 				if (_collBody.segmentIntersect(obj, seg, _collBody.get_currentState())){
214 					if (obj.fracOut < out.fracOut){
215 						out.posOut = obj.posOut;
216 						out.normalOut = obj.normalOut;
217 						out.fracOut = obj.fracOut;
218 						out.bodyOut = _collBody;
219 					}
220 				}
221 			}
222 		}
223 
224 		if (out.fracOut > 1)
225 			return false;
226 		
227 		if (out.fracOut < 0)
228 			out.fracOut = 0;
229 		else if (out.fracOut > 1)
230 			out.fracOut = 1;
231 		
232 		return true;
233 	};

235 	/**
236 	 * @function segmentBounding
237 	 * @param {JSegment} seg
238 	 * @param {RigidBody} obj
239 	 * @type boolean
240 	 **/
241 	CollisionSystem.prototype.segmentBounding=function(seg, obj){
242 		var pos = seg.getPoint(0.5);
243 		var r = Vector3DUtil.get_length(seg.delta) / 2;
244 
245 		if (obj.get_type() != "PLANE" && obj.get_type() != "TERRAIN"){
246 			var num1 = Vector3DUtil.get_length(Vector3DUtil.subtract(pos, obj.get_currentState().position));
247 			var num2 = r + obj.get_boundingSphere();
248 			if (num1 <= num2){
249 				return true;
250 			}else{
251 				return false;
252 			}
253 		}else{
254 			return true;
255 		}
256 	};

258 	/**
259 	 * @function findBody determines if a given rigid body is registered with the collision system
260 	 * @param {RigidBody} body
261 	 * @returns true if the body is registered, false if not
262 	 * @type boolean
263 	 **/
264 	CollisionSystem.prototype.findBody=function(body){
265 		for(var i=0, cbl=this.collBody.length; i<cbl; i++){
266 			var _collBody=this.collBody[i];
267 			if (body == _collBody)
268 				return true;
269 		}
270 		return false;
271 	};

274 	 * @function checkCollidables
275 	 * @param {RigidBody} body0
276 	 * @param {RigidBody} body1
277 	 * @type boolean
278 	 **/
279 	CollisionSystem.prototype.checkCollidables=function(body0, body1){
280 		if (body0.get_nonCollidables().length == 0 && body1.get_nonCollidables().length == 0){
281 			return true;
282 		}
283 		var nonCollidables=body0.get_nonCollidables();
284 		for(var i=0, ncl=nonCollidables.length; i<ncl; i++){
285 			var _body0=nonCollidables[i];
286 			if (body1 == _body0)
287 				return false;
288 		}
289 		nonCollidables=body1.get_nonCollidables();
290 		for(var i=0, ncl=nonCollidables.length; i<ncl; i++){
291 			var _body1=nonCollidables[i];
292 			if (body0 == _body1)
293 				return false;
294 		}
295 		return true;
296 	};
