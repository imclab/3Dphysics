 45 	JOctree.prototype.addTriangles=function(vertices, numVertices, triangleVertexIndices, numTriangles){
 46 		this.clear();
 47                         
 48 		this._vertices.concat(vertices);
 49                         
 50 		var NLen,tiny=JNumber3D.NUM_TINY;
 51 		var i0,i1,i2;
 52 		var dr1,dr2,N;
 53 		var indexedTriangle;
 54 		for(var i=0;i<triangleVertexIndices.length;i++){
 55 			var tri=triangleVertexIndices[i];

 62 			N = Vector3DUtil.crossProduct(dr1,dr2);
 63 			NLen = Vector3DUtil.get_length(N);
 64                                 
 65 			if (NLen > tiny){
 66 				indexedTriangle = new JIndexedTriangle();
 67 				indexedTriangle.setVertexIndices(i0, i1, i2, this._vertices);
 68 				this._triangles.push(indexedTriangle);
 69 			}
 70 		}

 74 	/* Builds the octree from scratch (not incrementally) - deleting
 75                  any previous tree.  Building the octree will involve placing
 76                  all triangles into the root cell.  Then this cell gets pushed
 77                  onto a stack of cells to examine. This stack will get parsed
 78                  and every cell containing more than maxTrianglesPerCell will
 79                  get split into 8 children, and all the original triangles in
 80                  that cell will get partitioned between the children. A
 81                  triangle can end up in multiple cells (possibly a lot!) if it
 82                  straddles a boundary. Therefore when intersection tests are
 83                  done tIndexedTriangle::m_counter can be set/tested using a
 84                  counter to avoid properly testing the triangle multiple times
 85                  (the counter _might_ wrap around, so when it wraps ALL the
 86                  triangle flags should be cleared! Could do this
 87                  incrementally...).*/
 88 	JOctree.prototype.buildOctree=function(maxTrianglesPerCell, minCellSize){
 89 		this._boundingBox.clear();
 90                         
 91 		for(var i=0;i<this._vertices.length;i++){
 92 			var vt=this._vertices[i];
 93 			this._boundingBox.addPoint(vt);
 94 		}
 95                         
 96 		this._cells=[];
 97 		this._cells.push(new OctreeCell(this._boundingBox));
 98                         
 99 		var numTriangles = this._triangles.length;
100 		for (var i = 0; i < numTriangles; i++ ) {
101 			this._cells[0].triangleIndices[i] = i;
102 		}
103                         
104 		var cellsToProcess = [];
105 		cellsToProcess.push(0);
106                         
107 		var iTri;
108 		var cellIndex;
109 		var childCell;
110 		while (cellsToProcess.length != 0) {
111 			cellIndex = cellsToProcess.pop();
112 			if (this._cells[cellIndex].triangleIndices.length <= maxTrianglesPerCell || this._cells[cellIndex].AABox.getRadiusAboutCentre() < minCellSize) {
113 				continue;
114 			}
115 			for (i = 0; i < OctreeCell.NUM_CHILDREN; i++ ) {
116 				this._cells[cellIndex].childCellIndices[i] = this._cells.length;
117 				cellsToProcess.push(this._cells.length);
118 				this._cells.push(new OctreeCell(this.createAABox(this._cells[cellIndex].AABox, i)));
119                                         
120 				childCell = this._cells[this._cells.length - 1];
121 				numTriangles = this._cells[cellIndex].triangleIndices.length;
122 				for (var j=0; j < numTriangles; j++ ) {
123 					iTri = this._cells[cellIndex].triangleIndices[j];
124 					if (this.doesTriangleIntersectCell(this._triangles[iTri], childCell))
125 					{
126 						childCell.triangleIndices.push(iTri);
127 					}
128 				}
129 			}
130 			this._cells[cellIndex].triangleIndices=[];
131 		}
132 	}

142 	/* Gets a list of all triangle indices that intersect an AABox. The vector passed in resized,
143                  so if you keep it between calls after a while it won't grow any more, and this
144                  won't allocate more memory.
145                  Returns the number of triangles (same as triangles.size())*/
146 	JOctree.prototype.getTrianglesIntersectingtAABox=function(triangles, aabb){
147 		if (this._cells.length == 0) return 0;
148                         
149 		this._cellsToTest=[];
150 		this._cellsToTest.push(0);
151                         
152 		this.incrementTestCounter();
153                         
154 		var cellIndex,nTris,cell,triangle;
155                         
156 		while (this._cellsToTest.length != 0) {
157 			cellIndex = this._cellsToTest.pop();
158                                 
159 			cell = this._cells[cellIndex];
160                                 
161 			if (!aabb.overlapTest(cell.AABox)) {
162 				continue;
163 			}
164                                 
165 			if (cell.isLeaf()) {
166 				nTris = cell.triangleIndices.length;
167 				for (var i = 0 ; i < nTris ; i++) {
168 					triangle = this.getTriangle(cell.triangleIndices[i]);
169 					if (triangle.counter != this._testCounter) {
170 						triangle.counter = this._testCounter;
171 						if (aabb.overlapTest(triangle.get_boundingBox())) {
172 							triangles.push(cell.triangleIndices[i]);
173 						}
174 					}
175 				}
176 			}else {
177 				for (i = 0 ; i < OctreeCell.NUM_CHILDREN ; i++) {
178 					this._cellsToTest.push(cell.childCellIndices[i]);
179 				}
180 			}
181 		}
182 		return triangles.length;
183 	}

212 	// Create a bounding box appropriate for a child, based on a parents AABox
213 	JOctree.prototype.createAABox=function(aabb, _id){
214 		var dims = JNumber3D.getScaleVector(Vector3DUtil.subtract(aabb.maxPos,aabb.minPos), 0.5);
215 		var offset;
216 		switch(_id) {
217 			case 0:
218 				offset = [1, 1, 1];
219 				break;
220 			case 1:
221 				offset = [1, 1, 0];
222 				break;
223 			case 2:
224 				offset = [1, 0, 1];
225 				break;
226 			case 3:
227 				offset = [1, 0, 0];
228 				break;
229 			case 4:
230 				offset = [0, 1, 1];
231 				break;
232 			case 5:
233 				offset = [0, 1, 0];
234 				break;
235 			case 6:
236 				offset = [0, 0, 1];
237 				break;
238 			case 7:
239 				offset = [0, 0, 0];
240 				break;
241 			default:
242 				offset = [0, 0, 0];
243 				break;
244 		}
245                         
246 		var result = new JAABox();
247 		result.minPos = Vector3DUtil.add(aabb.minPos,[offset.x * dims.x, offset.y * dims.y, offset.z * dims.z]);
248 		result.maxPos = Vector3DUtil.add(result.minPos,dims);
249                         
250 		Vector3DUtil.scaleBy(dims,0.00001);
251 		result.minPos = Vector3DUtil.subtract(result.minPos,dims);
252 		result.maxPos = Vector3DUtil.subtract(result.maxPos,dims);
253                         
254 		return result;
255 	}

258 	// Returns true if the triangle intersects or is contained by a cell
259 	JOctree.prototype.doesTriangleIntersectCell=function(triangle, cell){
260 		if (!triangle.get_boundingBox().overlapTest(cell.AABox)) {
261 			return false;
262 		}
263 		if (cell.AABox.isPointInside(this.getVertex(triangle.getVertexIndex(0))) ||
264 			cell.AABox.isPointInside(this.getVertex(triangle.getVertexIndex(1))) ||
265 			cell.AABox.isPointInside(this.getVertex(triangle.getVertexIndex(2)))) {
266 			return true;
267 		}
268                                 
269 		var tri = new JTriangle(this.getVertex(triangle.getVertexIndex(0)), this.getVertex(triangle.getVertexIndex(1)), this.getVertex(triangle.getVertexIndex(2)));
270 		var edge;
271 		var seg;
272 		var edges = cell.egdes;
273 		var pts = cell.points;
274 		for (var i = 0; i < 12; i++ ) {
275 			edge = edges[i];
276 			seg = new JSegment(pts[edge.ind0], Vector3DUtil.subtract(pts[edge.ind1],pts[edge.ind0]));
277 			if (tri.segmentTriangleIntersection(null, seg)) {
278 				return true;
279 			}
280 		}
281                         
282 		var pt0;
283 		var pt1;
284 		for (i = 0; i < 3; i++ ) {
285 			pt0 = tri.getVertex(i);
286 			pt1 = tri.getVertex((i + 1) % 3);
287 			if (cell.AABox.segmentAABoxOverlap(new JSegment(pt0, Vector3DUtil.subtract(pt1,pt0)))) {
288 				return true;
289 			}
290 		}
291 		return false;
292 	}

295 	/* Increment our test counter, wrapping around if necessary and zapping the triangle counters.
296                  Const because we only modify mutable members.*/
297 	JOctree.prototype.incrementTestCounter=function(){
298 		++this._testCounter;
299 		if (this._testCounter == 0) {
300 			var numTriangles = this._triangles.length;
301 			for (var i = 0; i < numTriangles; i++) {
302 				this._triangles[i].counter = 0;
303 			}
304 			this._testCounter = 1;
305 		}
306 	}
