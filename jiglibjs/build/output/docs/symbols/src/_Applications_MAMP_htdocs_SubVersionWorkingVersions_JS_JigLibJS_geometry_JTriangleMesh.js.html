 69 		var segBox = new JAABox();
 70 		segBox.addSegment(seg);
 71                         
 72 		var potentialTriangles = [];
 73 		var numTriangle = this._octree.getTrianglesIntersectingtAABox(potentialTriangles, segBox);
 74                         
 75 		var bestFrac = JNumber3D.NUM_HUGE;
 76 		var tri;
 77 		var meshTriangle;
 78 		for (var iTriangle = 0 ; iTriangle < numTriangles; iTriangle++) {
 79 			meshTriangle = this._octree.getTriangle(potentialTriangles[iTriangle]);
 80                                 
 81 			tri = new JTriangle(this._octree.getVertex(meshTriangle.getVertexIndex(0)), this._octree.getVertex(meshTriangle.getVertexIndex(1)), this._octree.getVertex(meshTriangle.getVertexIndex(2)));
 82                                 
 83 			if (tri.segmentTriangleIntersection(out, seg)) {
 84 				if (out.frac < bestFrac) {
 85 					bestFrac = out.frac;
 86 					out.position = seg.getPoint(bestFrac);
 87 					out.normal = meshTriangle.plane.normal;
 88 				}
 89 			}
 90 		}
 91 		out.frac = bestFrac;
 92 		if (bestFrac < JNumber3D.NUM_HUGE) {
 93 			return true;
 94 		}else {
