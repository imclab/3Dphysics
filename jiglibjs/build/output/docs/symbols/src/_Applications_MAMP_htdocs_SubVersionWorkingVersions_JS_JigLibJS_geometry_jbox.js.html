131 	/**
132 	 * @function getSpan returns the minimum and maximum extents of the box along the axis, relative to the center of the box.
133 	 * @param {array} axis the axis expressed as a 3D vector
134 	 * @type SpanData
135 	 **/
136 	JBox.prototype.getSpan=function(axis){
137 		var cols= this.get_currentState().getOrientationCols();
138 		var obj = new SpanData();
139 		var s = Math.abs(Vector3DUtil.dotProduct(axis, cols[0])) * (0.5 * this._sideLengths[0]);
140 		var u = Math.abs(Vector3DUtil.dotProduct(axis, cols[1])) * (0.5 * this._sideLengths[1]);
141 		var d = Math.abs(Vector3DUtil.dotProduct(axis, cols[2])) * (0.5 * this._sideLengths[2]);
142 		var r = s + u + d;
143 		var p = Vector3DUtil.dotProduct(this.get_currentState().position, axis);
144 		obj.min = p - r;
145 		obj.max = p + r;
146 
147 		return obj;
148 	};

174 	 * @function getCornerPointsInBoxSpace returns the corner points of this JBox in another box space
175 	 * @param {PhysicsState} thisState
176 	 * @param {PhysicsState} boxState
177 	 * @type array

179 	JBox.prototype.getCornerPointsInBoxSpace=function(thisState, boxState){
180 		var max = JMatrix3D.getTransposeMatrix(boxState.get_orientation());
181 		var pos = Vector3DUtil.subtract(thisState.position,boxState.position);
182 		JMatrix3D.multiplyVector(max, pos);
183 						
184 		var orient = JMatrix3D.getAppendMatrix3D(thisState.get_orientation(), max);
185 						
186 		var arr = [];
187 						
188 		var transform = JMatrix3D.getTranslationMatrix(pos[0], pos[1], pos[2]);
189 		transform = JMatrix3D.getAppendMatrix3D(orient, transform);
190 		
191 		for(var i=0;i<this._points.length;i++){
192 			_point=this._points[i].slice(0);

199 	/**
200 	 * @function getSqDistanceToPoint
201 	 * @param {PhysicsState} state
202 	 * @param {array} closestBoxPoint
203 	 * @param {array} point
204 	 * @type number
205 	 **/
206 	JBox.prototype.getSqDistanceToPoint=function(state, closestBoxPoint, point){
207 		closestBoxPoint.pos = Vector3DUtil.subtract(point, state.position);
208 		JMatrix3D.multiplyVector(JMatrix3D.getTransposeMatrix(state.get_orientation()), closestBoxPoint.pos);
209 
210 		var delta = 0;
211 		var sqDistance = 0;
212 		var halfSideLengths = this.getHalfSideLengths();
213 
214 		if (closestBoxPoint.pos[0] < -halfSideLengths[0]){
215 			delta = closestBoxPoint.pos[0] + halfSideLengths[0];
216 			sqDistance += (delta * delta);
217 			closestBoxPoint.pos[0] = -halfSideLengths[0];
218 		}else if (closestBoxPoint.pos[0] > halfSideLengths[0]){
219 			delta = closestBoxPoint.pos[0] - halfSideLengths[0];
220 			sqDistance += (delta * delta);
221 			closestBoxPoint.pos[0] = halfSideLengths[0];
222 		}
223 
224 		if (closestBoxPoint.pos[1] < -halfSideLengths[1]){
225 			delta = closestBoxPoint.pos[1] + halfSideLengths[1];
226 			sqDistance += (delta * delta);
227 			closestBoxPoint.pos[1] = -halfSideLengths[1];
228 		}else if (closestBoxPoint.pos[1] > halfSideLengths[1]){
229 			delta = closestBoxPoint.pos[1] - halfSideLengths[1];
230 			sqDistance += (delta * delta);
231 			closestBoxPoint.pos[1] = halfSideLengths[1];
232 		}
233 
234 		if (closestBoxPoint.pos[2] < -halfSideLengths[2]){
235 			delta = closestBoxPoint.pos[2] + halfSideLengths[2];
236 			sqDistance += (delta * delta);
237 			closestBoxPoint.pos[2] = -halfSideLengths[2];
238 		}else if (closestBoxPoint.pos[2] > halfSideLengths[2]){
239 			delta = (closestBoxPoint.pos[2] - halfSideLengths[2]);
240 			sqDistance += (delta * delta);
241 			closestBoxPoint.pos[2] = halfSideLengths[2];
242 		}
243 		JMatrix3D.multiplyVector(state.get_orientation(), closestBoxPoint.pos);
244 		closestBoxPoint.pos = Vector3DUtil.add(state.position, closestBoxPoint.pos);
245 		return sqDistance;
246 	};

260 	 * @function pointIntersect 
261 	 * @param {array} pos
262 	 * @type boolean
263 	 **/
264 	JBox.prototype.pointIntersect=function(pos){
265 		var p = Vector3DUtil.subtract(pos, this.get_currentState().position);
266 		var h = JNumber3D.getScaleVector(this._sideLengths, 0.5);
267 		var dirVec;
268 		var cols = this.get_currentState().getOrientationCols();
269 		for (var dir; dir < 3; dir++){
270 			dirVec = cols[dir].slice(0);
271 			Vector3DUtil.normalize(dirVec);
272 			if (Math.abs(Vector3DUtil.dotProduct(dirVec, p)) > h[dir] + JNumber3D.NUM_TINY){
273 				return false;
274 			}
275 		}
276 		return true;
277 	};

279 	/**
280 	 * @function getSupportVertices 
281 	 * @param {array} axis
282 	 * @type array
283 	 **/
284 	JBox.prototype.getSupportVertices=function(axis){
285 		var vertices = [];
286 		var d = [1,1,1];
287 		var H;
288 		var temp = this.get_currentState().getOrientationCols();
289 		Vector3DUtil.normalize(temp[0]);
290 		Vector3DUtil.normalize(temp[1]);
291 		Vector3DUtil.normalize(temp[2]);
292 		for (var i = 0; i < 3; i++){
293 			d[i] = Vector3DUtil.dotProduct(axis, temp[i]);
294 			if (Math.abs(d[i]) > 1 - 0.001){
295 				var f = (d[i] < 0) ? (i * 2) : (i * 2) + 1;
296 				for (var j = 0; j < 4; j++){
297 					H = this._points[this._faces[f][j]];
298 					var _vj = vertices[j] = this.get_currentState().position.slice(0);
299 					_vj = Vector3DUtil.add(_vj, JNumber3D.getScaleVector(temp[0], H[0]));
300 					_vj = Vector3DUtil.add(_vj, JNumber3D.getScaleVector(temp[1], H[1]));
301 					_vj = Vector3DUtil.add(_vj, JNumber3D.getScaleVector(temp[2], H[2]));
302 				}
303 				return vertices;
304 			}
305 		}
306 
307 		for (i = 0; i < 3; i++){
308 			if (Math.abs(d[i]) < 0.005){
309 				var k;
310 				var m = (i + 1) % 3;
311 				var n = (i + 2) % 3;
312 
313 				H = this.get_currentState().position.slice(0);
314 				k = (d[m] > 0) ? -1 : 1;
315 				H = Vector3DUtil.add(H, JNumber3D.getScaleVector(temp[m], k * this._sideLengths[m] / 2));
316 				k = (d[n] > 0) ? -1 : 1;
317 				H = Vector3DUtil.add(H, JNumber3D.getScaleVector(temp[n], k * this._sideLengths[n] / 2));
318 
319 				vertices[0] = Vector3DUtil.add(H, JNumber3D.getScaleVector(temp[i], this._sideLengths[i] / 2));
320 				vertices[1] = Vector3DUtil.add(H, JNumber3D.getScaleVector(temp[i], -this._sideLengths[i] / 2));
321 				return vertices;
322 			}
323 		}
324 
325 		var _v0 =vertices[0] = this.get_currentState().position.slice(0);
326 		k = (d[0] > 0) ? -1 : 1;
327 		vertices[0] = Vector3DUtil.add(_v0, JNumber3D.getScaleVector(temp[0], k * this._sideLengths[0] / 2));
328 		k = (d[1] > 0) ? -1 : 1;
329 		vertices[0] = Vector3DUtil.add(_v0, JNumber3D.getScaleVector(temp[1], k * this._sideLengths[1] / 2));
330 		k = (d[2] > 0) ? -1 : 1;
331 		vertices[0] = Vector3DUtil.add(_v0, JNumber3D.getScaleVector(temp[2], k * this._sideLengths[2] / 2));
332 		return vertices;
333 	};

337 	 * @function segmentIntersect 
338 	 * @param {object} out
339 	 * @param {JSegment} seg
340 	 * @param {PhysicsState} state
341 	 * @type boolean
342 	 **/
343 	JBox.prototype.segmentIntersect=function(out, seg, state){
344 		out.fracOut = 0;
345 		out.posOut = [0,0,0,0];
346 		out.normalOut = [0,0,0,0];
347 
348 		var frac = JNumber3D.NUM_HUGE;
349 		var min = -JNumber3D.NUM_HUGE;
350 		var max = JNumber3D.NUM_HUGE;
351 		var dirMin = 0;
352 		var dirMax = 0;
353 		var dir = 0;
354 		var p = Vector3DUtil.subtract(state.position, seg.origin);
355 		var h = JNumber3D.getScaleVector(this._sideLengths, 0.5);
356 
357 		//var tempV:Vector3D;
358 		var e;
359 		var f;
360 		var t;
361 		var t1;
362 		var t2;
363 						
364 		var orientationCol = state.getOrientationCols();
365 		var directionVectorArray = h.slice(0);
366 		var directionVectorNumber;
367 		for (dir = 0; dir < 3; dir++){
368 			directionVectorNumber = directionVectorArray[dir];
369 			e = Vector3DUtil.dotProduct(orientationCol[dir], p);
370 			f = Vector3DUtil.dotProduct(orientationCol[dir], seg.delta);
371 			if (Math.abs(f) > JNumber3D.NUM_TINY){
372 				t1 = (e + directionVectorNumber) / f;
373 				t2 = (e - directionVectorNumber) / f;
374 				if (t1 > t2){
375 					t = t1;
376 					t1 = t2;
377 					t2 = t;
378 				}
379 				if (t1 > min){
380 					min = t1;
381 					dirMin = dir;
382 				}
383 				if (t2 < max){
384 					max = t2;
385 					dirMax = dir;
386 				}
387 				if (min > max) return false;
388 				if (max < 0) return false;
389 			}else if (-e - directionVectorNumber > 0 || -e + directionVectorNumber < 0){
390 				return false;
391 			}
392 		}
393 
394 		if (min > 0){
395 			dir = dirMin;
396 			frac = min;
397 		}else{
398 			dir = dirMax;
399 			frac = max;
400 		}
401 		if (frac < 0) frac = 0;
402 		/*if (frac > 1)
403 		frac = 1;*/
404 		if (frac > 1 - JNumber3D.NUM_TINY){
405 			return false;
406 		}
407 		out.fracOut = frac;
408 		out.posOut = seg.getPoint(frac);
409 
410 		if (Vector3DUtil.dotProduct(orientationCol[dir], seg.delta) < 0)
411 			out.normalOut = JNumber3D.getScaleVector(orientationCol[dir], -1);
412 		else
413 			out.normalOut = orientationCol[dir];
414 
415 		return true;
416 	};
