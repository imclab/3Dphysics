 73 	/**
 74 	 * @function segmentSegmentDistanceSq
 75 	 * @param {object} out
 76 	 * @param {JSegment} seg
 77 	 * @type number
 78 	 **/
 79 	JSegment.prototype.segmentSegmentDistanceSq=function(out, seg){
 80 		out.t0 = 0;
 81 		out.t1 = 0;
 82 
 83 		var kDiff = Vector3DUtil.subtract(this.origin, seg.origin);
 84 		var fA00 = Vector3DUtil.get_lengthSquared(this.delta);
 85 		var fA01 = -Vector3DUtil.dotProduct(this.delta, seg.delta);
 86 		var fA11 = Vector3DUtil.get_lengthSquared(seg.delta);
 87 		var fB0 = Vector3DUtil.dotProduct(kDiff, this.delta);
 88 		var fC = Vector3DUtil.get_lengthSquared(kDiff);
 89 		var fDet = Math.abs(fA00 * fA11 - fA01 * fA01);
 90 		var fB1;
 91 		var fS;
 92 		var fT;
 93 		var fSqrDist;
 94 		var fTmp;
 95 
 96 		if (fDet >= JNumber3D.NUM_TINY){
 97 			fB1 = -Vector3DUtil.dotProduct(kDiff, seg.delta);
 98 			fS = fA01 * fB1 - fA11 * fB0;
 99 			fT = fA01 * fB0 - fA00 * fB1;
100 
101 			if (fS >= 0){
102 				if (fS <= fDet){
103 					if (fT >= 0){
104 						if (fT <= fDet){
105 							var fInvDet = 1 / fDet;
106 							fS *= fInvDet;
107 							fT *= fInvDet;
108 							fSqrDist = fS * (fA00 * fS + fA01 * fT + 2 * fB0) + fT * (fA01 * fS + fA11 * fT + 2 * fB1) + fC;
109 						}else{
110 							fT = 1;
111 							fTmp = fA01 + fB0;
112 							if (fTmp >= 0){
113 								fS = 0;
114 								fSqrDist = fA11 + 2 * fB1 + fC;
115 							}else if (-fTmp >= fA00){
116 								fS = 1;
117 								fSqrDist = fA00 + fA11 + fC + 2 * (fB1 + fTmp);
118 							}else{
119 								fS = -fTmp / fA00;
120 								fSqrDist = fTmp * fS + fA11 + 2 * fB1 + fC;
121 							}
122 						}
123 					}else{
124 						fT = 0;
125 						if (fB0 >= 0){
126 							fS = 0;
127 							fSqrDist = fC;
128 						}else if (-fB0 >= fA00){
129 							fS = 1;
130 							fSqrDist = fA00 + 2 * fB0 + fC;
131 						}else{
132 							fS = -fB0 / fA00;
133 							fSqrDist = fB0 * fS + fC;
134 						}
135 					}
136 				}else{
137 					if (fT >= 0){
138 						if (fT <= fDet){
139 							fS = 1;
140 							fTmp = fA01 + fB1;
141 							if (fTmp >= 0){
142 								fT = 0;
143 								fSqrDist = fA00 + 2 * fB0 + fC;
144 							}else if (-fTmp >= fA11){
145 								fT = 1;
146 								fSqrDist = fA00 + fA11 + fC + 2 * (fB0 + fTmp);
147 							}else{
148 								fT = -fTmp / fA11;
149 								fSqrDist = fTmp * fT + fA00 + 2 * fB0 + fC;
150 							}
151 						}else{
152 							fTmp = fA01 + fB0;
153 							if (-fTmp <= fA00){
154 								fT = 1;
155 								if (fTmp >= 0){
156 									fS = 0;
157 									fSqrDist = fA11 + 2 * fB1 + fC;
158 								}else{
159 									fS = -fTmp / fA00;
160 									fSqrDist = fTmp * fS + fA11 + 2 * fB1 + fC;
161 								}
162 							}else{
163 								fS = 1;
164 								fTmp = fA01 + fB1;
165 								if (fTmp >= 0){
166 									fT = 0;
167 									fSqrDist = fA00 + 2 * fB0 + fC;
168 								}else if (-fTmp >= fA11){
169 									fT = 1;
170 									fSqrDist = fA00 + fA11 + fC + 2 * (fB0 + fTmp);
171 								}else{
172 									fT = -fTmp / fA11;
173 									fSqrDist = fTmp * fT + fA00 + 2 * fB0 + fC;
174 								}
175 							}
176 						}
177 					}else{
178 						if (-fB0 < fA00){
179 							fT = 0;
180 							if (fB0 >= 0){
181 								fS = 0;
182 								fSqrDist = fC;
183 							}else{
184 								fS = -fB0 / fA00;
185 								fSqrDist = fB0 * fS + fC;
186 							}
187 						}else{
188 							fS = 1;
189 							fTmp = fA01 + fB1;
190 							if (fTmp >= 0){
191 								fT = 0;
192 								fSqrDist = fA00 + 2 * fB0 + fC;
193 							}else if (-fTmp >= fA11){
194 								fT = 1;
195 								fSqrDist = fA00 + fA11 + fC + 2 * (fB0 + fTmp);
196 							}else{
197 								fT = -fTmp / fA11;
198 								fSqrDist = fTmp * fT + fA00 + 2 * fB0 + fC;
199 							}
200 						}
201 					}
202 				}
203 			}else{
204 				if (fT >= 0){
205 					if (fT <= fDet){
206 						fS = 0;
207 						if (fB1 >= 0){
208 							fT = 0;
209 							fSqrDist = fC;
210 						}else if (-fB1 >= fA11){
211 							fT = 1;
212 							fSqrDist = fA11 + 2 * fB1 + fC;
213 						}else{
214 							fT = -fB1 / fA11;
215 							fSqrDist = fB1 * fT + fC;
216 						}
217 					}else{
218 						fTmp = fA01 + fB0;
219 						if (fTmp < 0){
220 							fT = 1;
221 							if (-fTmp >= fA00){
222 								fS = 1;
223 								fSqrDist = fA00 + fA11 + fC + 2 * (fB1 + fTmp);
224 							}else{
225 								fS = -fTmp / fA00;
226 								fSqrDist = fTmp * fS + fA11 + 2 * fB1 + fC;
227 							}
228 						}else{
229 							fS = 0;
230 							if (fB1 >= 0){
231 								fT = 0;
232 								fSqrDist = fC;
233 							}else if (-fB1 >= fA11){
234 								fT = 1;
235 								fSqrDist = fA11 + 2 * fB1 + fC;
236 							}else{
237 								fT = -fB1 / fA11;
238 								fSqrDist = fB1 * fT + fC;
239 							}
240 						}
241 					}
242 				}else{
243 					if (fB0 < 0){
244 						fT = 0;
245 						if (-fB0 >= fA00){
246 							fS = 1;
247 							fSqrDist = fA00 + 2 * fB0 + fC;
248 						}else{
249 							fS = -fB0 / fA00;
250 							fSqrDist = fB0 * fS + fC;
251 						}
252 					}else{
253 						fS = 0;
254 						if (fB1 >= 0){
255 							fT = 0;
256 							fSqrDist = fC;
257 						}else if (-fB1 >= fA11){
258 							fT = 1;
259 							fSqrDist = fA11 + 2 * fB1 + fC;
260 						}else{
261 							fT = -fB1 / fA11;
262 							fSqrDist = fB1 * fT + fC;
263 						}
264 					}
265 				}
266 			}
267 		}else{
268 			if (fA01 > 0){
269 				if (fB0 >= 0){
270 					fS = 0;
271 					fT = 0;
272 					fSqrDist = fC;
273 				}else if (-fB0 <= fA00){
274 					fS = -fB0 / fA00;
275 					fT = 0;
276 					fSqrDist = fB0 * fS + fC;
277 				}else{
278 					fB1 = -Vector3DUtil.dotProduct(kDiff, seg.delta);
279 					fS = 1;
280 					fTmp = fA00 + fB0;
281 					if (-fTmp >= fA01){
282 						fT = 1;
283 						fSqrDist = fA00 + fA11 + fC + 2 * (fA01 + fB0 + fB1);
284 					}else{
285 						fT = -fTmp / fA01;
286 						fSqrDist = fA00 + 2 * fB0 + fC + fT * (fA11 * fT + 2 * (fA01 + fB1));
287 					}
288 				}
289 			}else{
290 				if (-fB0 >= fA00){
291 					fS = 1;
292 					fT = 0;
293 					fSqrDist = fA00 + 2 * fB0 + fC;
294 				}else if (fB0 <= 0) {
295 					fS = -fB0 / fA00;
296 					fT = 0;
297 					fSqrDist = fB0 * fS + fC;
298 				}else{
299 					fB1 = -Vector3DUtil.dotProduct(kDiff, seg.delta);
300 					fS = 0;
301 					if (fB0 >= -fA01){
302 						fT = 1;
303 						fSqrDist = fA11 + 2 * fB1 + fC;
304 					}else{
305 						fT = -fB0 / fA01;
306 						fSqrDist = fC + fT * (2 * fB1 + fA11 * fT);
307 					}
308 				}
309 			}
310 		}
311 
312 		out.t0 = fS;
313 		out.t1 = fT;
314 		return Math.abs(fSqrDist);
315 	};

317 	/**
318 	 * @function pointSegmentDistanceSq
319 	 * @param {object} out
320 	 * @param {array} pt
321 	 * @type number
322 	 **/
323 	JSegment.prototype.pointSegmentDistanceSq=function(out, pt){
324 		out.t = 0;
325 
326 		var kDiff = Vector3DUtil.subtract(pt,  this.origin);
327 		var fT = Vector3DUtil.dotProduct(kDiff, this.delta);
328 
329 		if (fT <= 0){
330 			fT = 0;
331 		}else{
332 			var fSqrLen = Vector3DUtil.get_lengthSquared(this._delta);
333 			if (fT >= fSqrLen){
334 				fT = 1;
335 				kDiff = Vector3DUtil.subtract(kDiff, this._delta);
336 			}else{
337 				fT /= fSqrLen;
338 				kDiff = Vector3DUtil.subtract(kDiff, JNumber3D.getScaleVector(this._delta, fT));
339 			}
340 		}
341 
342 		out.t = fT;
343 		return Vector3DUtil.get_lengthSquared(kDiff);
344 	};

381 	/**
382 	 * @function sqrDistanceLine
383 	 * @param {object} out
384 	 * @param {JRay} rkLine
385 	 * @param {JBox} rkBox
386 	 * @param {PhysicsState} boxState
387 	 * @type number
388 	 **/
389 	JSegment.prototype.sqrDistanceLine=function(out, rkLine, rkBox, boxState){
390 		var orientationCols = boxState.getOrientationCols();
391 		out.num = 0;
392 		out.num0 = 0;
393 		out.num1 = 0;
394 		out.num2 = 0;
395 
396 		var kDiff = Vector3DUtil.subtract(rkLine.origin, boxState.position);
397 		var kPnt = Vector3DUtil.create( Vector3DUtil.dotProduct(kDiff, orientationCols[0]),
398 										Vector3DUtil.dotProduct(kDiff, orientationCols[1]),
399 										Vector3DUtil.dotProduct(kDiff, orientationCols[2]), 
400 										0);
401 
402 		var kDir = Vector3DUtil.create( Vector3DUtil.dotProduct(rkLine.dir, orientationCols[0]),
403 										Vector3DUtil.dotProduct(rkLine.dir, orientationCols[1]),
404 										Vector3DUtil.dotProduct(rkLine.dir, orientationCols[2]), 
405 							            0);
406 						
407 		var kPntArr = kPnt.slice(0);
408 		var kDirArr = kDir.slice(0);
409 						
410 		var bReflect = [1,1,1,0];
411 		for (var i = 0; i < 3; i++){
412 			if (kDirArr[i] < 0){
413 				kPntArr[i] = -kPntArr[i];
414 				kDirArr[i] = -kDirArr[i];
415 				bReflect[i] = true;
416 			}else{
417 				bReflect[i] = false;
418 			}
419 		}
420 
421 		JNumber3D.copyFromArray(kPnt, kPntArr);
422 		JNumber3D.copyFromArray(kDir, kDirArr);
423 						
424 		var obj = {};
425 		obj.rkPnt = kPnt.slice(0);
426 		obj.pfLParam = 0;
427 		obj.rfSqrDistance = 0;
428 
429 		if (kDir[0] > 0){
430 			if (kDir[1] > 0){
431 				if (kDir[2] > 0){
432 					this.caseNoZeros(obj, kDir, rkBox);
433 					out.num = obj.pfLParam;
434 				}else{
435 					this.case0(obj, 0, 1, 2, kDir, rkBox);
436 					out.num = obj.pfLParam;
437 				}
438 			}else{
439 				if (kDir[2] > 0){
440 					this.case0(obj, 0, 2, 1, kDir, rkBox);
441 					out.num = obj.pfLParam;
442 				}else{
443 					this.case00(obj, 0, 1, 2, kDir, rkBox);
444 					out.num = obj.pfLParam;
445 				}
446 			}
447 		}else{
448 			if (kDir[1] > 0){
449 				if (kDir[2] > 0){
450 					this.case0(obj, 1, 2, 0, kDir, rkBox);
451 					out.num = obj.pfLParam;
452 				}else{
453 					this.case00(obj, 1, 0, 2, kDir, rkBox);
454 					out.num = obj.pfLParam;
455 				}
456 			}else{
457 				if (kDir[2] > 0){
458 					this.case00(obj, 2, 0, 1, kDir, rkBox);
459 					out.num = obj.pfLParam;
460 				}else{
461 					this.case000(obj, rkBox);
462 					out.num = 0;
463 				}
464 			}
465 		}
466 
467 		kPntArr = obj.rkPnt.slice(0);
468 		for (i = 0; i < 3; i++){
469 			if (bReflect[i]) kPntArr[i] = -kPntArr[i];
470 		}
471 		JNumber3D.copyFromArray(obj.rkPnt, kPntArr);
472 
473 		out.num0 = obj.rkPnt[0];
474 		out.num1 = obj.rkPnt[1];
475 		out.num2 = obj.rkPnt[2];
476 
477 		return Math.max(obj.rfSqrDistance, 0);
478 	};

480 	/**
481 	 * @function sqrDistancePoint
482 	 * @param {object} out
483 	 * @param {array} rkPoint
484 	 * @param {JBox} rkBox
485 	 * @param {PhysicsState} boxState
486 	 * @type number
487 	 **/
488 	JSegment.prototype.sqrDistancePoint=function(out, rkPoint, rkBox, boxState){
489 		var orientationVector = boxState.getOrientationCols();
490 		var kDiff = Vector3DUtil.subtract(rkPoint, boxState.position);
491 		var kClosest = Vector3DUtil.create( Vector3DUtil.dotProduct(kDiff, orientationVector[0]),
492 							                Vector3DUtil.dotProduct(kDiff, orientationVector[1]),
493 							                Vector3DUtil.dotProduct(kDiff, orientationVector[2]), 
494 							                0);
495 
496 		var fSqrDistance = 0;
497 		var fDelta;
498 		var boxHalfSide = rkBox.getHalfSideLengths();
499 
500 		if (kClosest[0] < -boxHalfSide[0]){
501 			fDelta = kClosest[0] + boxHalfSide[0];
502 			fSqrDistance += (fDelta * fDelta);
503 			kClosest[0] = -boxHalfSide[0];
504 		}else if (kClosest[0] > boxHalfSide[0]){
505 			fDelta = kClosest[0] - boxHalfSide[0];
506 			fSqrDistance += (fDelta * fDelta);
507 			kClosest[0] = boxHalfSide[0];
508 		}
509 
510 		if (kClosest[1] < -boxHalfSide[1]){
511 			fDelta = kClosest[1] + boxHalfSide[1];
512 			fSqrDistance += (fDelta * fDelta);
513 			kClosest[1] = -boxHalfSide[1];
514 		}else if (kClosest[1] > boxHalfSide[1]){
515 			fDelta = kClosest[1] - boxHalfSide[1];
516 			fSqrDistance += (fDelta * fDelta);
517 			kClosest[1] = boxHalfSide[1];
518 		}
519 
520 		if (kClosest[2] < -boxHalfSide[2]){
521 			fDelta = kClosest[2] + boxHalfSide[2];
522 			fSqrDistance += (fDelta * fDelta);
523 			kClosest[2] = -boxHalfSide[2];
524 		}else if (kClosest[2] > boxHalfSide[2]){
525 			fDelta = kClosest[2] - boxHalfSide[2];
526 			fSqrDistance += (fDelta * fDelta);
527 			kClosest[2] = boxHalfSide[2];
528 		}
529 
530 		out.pfLParam0 = kClosest[0];
531 		out.pfLParam1 = kClosest[1];
532 		out.pfLParam2 = kClosest[2];
533 
534 		return Math.max(fSqrDistance, 0);
535 	};

537 	/**
538 	 * @function face
539 	 * @param {object} out
540 	 * @param {number} i0
541 	 * @param {number} i1
542 	 * @param {number} i2
543 	 * @param {array} rkDir
544 	 * @param {JBox} rkBox
545 	 * @param {array} rkPmE
546 	 * @type void
547 	 **/
548 	JSegment.prototype.face=function(out, i0, i1, i2, rkDir, rkBox, rkPmE){
549 		var kPpE = [0,0,0,0];
550 		var fLSqr;
551 		var fInv;
552 		var fTmp;
553 		var fParam;
554 		var fT;
555 		var fDelta;
556 
557 		var boxHalfSide = rkBox.getHalfSideLengths();
558 		var boxHalfArr = boxHalfSide;
559 		var rkPntArr = out.rkPnt;
560 		var rkDirArr = rkDir;
561 		var kPpEArr = kPpE;
562 		var rkPmEArr = rkPmE;
563
