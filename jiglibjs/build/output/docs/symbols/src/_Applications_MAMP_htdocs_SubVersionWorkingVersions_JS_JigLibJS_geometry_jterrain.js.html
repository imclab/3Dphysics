 47 	/**
 48 	 * @function getHeightAndNormalByPoint
 49 	 * @param {array} point
 50 	 * @type object
 51 	 **/
 52 	JTerrain.prototype.getHeightAndNormalByPoint=function(point){
 53 		var w = this.limiteInt(point[0], this._terrain.get_minW(), this._terrain.get_maxW());
 54 		var h = this.limiteInt(point[2], this._terrain.get_minH(), this._terrain.get_maxH());
 55 						
 56 		var i0 = ((w - this._terrain.get_minW()) / this._terrain.get_dw())|0;
 57 		var j0 = ((h - this._terrain.get_minH()) / this._terrain.get_dh())|0;
 58 		i0 = this.limiteInt(i0, 0, this._terrain.get_sw());
 59 		j0 = this.limiteInt(j0, 0, this._terrain.get_sh());
 60 		
 61 		var i1 = i0 + 1;
 62 		var j1 = j0 + 1;
 63 		i1 = this.limiteInt(i1, 0, this._terrain.get_sw());
 64 		j1 = this.limiteInt(j1, 0, this._terrain.get_sh());
 65 						
 66 		var iFrac = 1 - (w - (i0 * this._terrain.get_dw() + this._terrain.get_minW())) / this._terrain.get_dw();
 67 		var jFrac = (h - (j0 * this._terrain.get_dh() + this._terrain.get_minH())) / this._terrain.get_dh();
 68 		iFrac = JNumber3D.getLimiteNumber(iFrac, 0, 1);
 69 		jFrac = JNumber3D.getLimiteNumber(jFrac, 0, 1);
 70 						
 71 		var h00 = this._terrain.get_heights(i0,j0);
 72 		var h01 = this._terrain.get_heights(i0,j1);
 73 		var h10 = this._terrain.get_heights(i1,j0);
 74 		var h11 = this._terrain.get_heights(i1,j1);
 75 						
 76 		var obj = { };
 77 		obj.height = 0;
 78 		obj.normal = [0,0,0,0];
 79 		var plane;
 80 		if (iFrac < jFrac || i0==i1 || j0 == j1){
 81 			obj.normal = Vector3DUtil.crossProduct( [0, h11 - h10, this._terrain.get_dh(), 0],
 82 													[this._terrain.get_dw(), h11 - h01, 0, 0]);
 83 			Vector3DUtil.normalize(obj.normal);
 84 								
 85 			plane = new PlaneData([(i1 * this._terrain.get_dw() + this._terrain.get_minW()), h11, (j1 * this._terrain.get_dh() + this._terrain.get_minH()), 0], 
 86 								  obj.normal);
 87 			obj.height = plane.pointPlaneDistance(point);
 88 		}else{
 89 			obj.normal = Vector3DUtil.crossProduct( [0, h01 - h00, this._terrain.get_dh(), 0], 
 90 													[this._terrain.get_dw(), h10 - h00, 0, 0]);
 91 			Vector3DUtil.normalize(obj.normal);
 92 								
 93 			plane = new PlaneData([(i0 * this._terrain.get_dw() + this._terrain.get_minW()), h00, (j0 * this._terrain.get_dh() + this._terrain.get_minH()), 0], 
 94 			                      obj.normal);
 95 			obj.height = plane.pointPlaneDistance(point);
 96 		}
 97 		return obj;
 98 	};

129 	 * @function segmentIntersect
130 	 * @param {object} out
131 	 * @param {JSegment} seg
132 	 * @param {PhysicsState} state
133 	 * @type array
134 	 **/
135 	JTerrain.prototype.segmentIntersect=function(out, seg, state){
136 		out.fracOut = 0;
137 		out.posOut = [0,0,0,0];
138 		out.normalOut = [0,0,0,0];
139 						
140 		if (seg.delta[1] > -JNumber3D.NUM_TINY) 
141 			return false;
142 
143 		var obj1 = this.getHeightAndNormalByPoint(seg.origin);
144 		if (obj1.height < 0) 
145 			return false;
146 
147 		var obj2 = getHeightAndNormalByPoint(seg.getEnd());
148 		if (obj2.height > 0) 
149 			return false;
150 
151 		var depthEnd = -obj2.height;
152 		var weightStart = 1 / (JNumber3D.NUM_TINY + obj1.height);
153 		var weightEnd = 1 / (JNumber3D.NUM_TINY + obj2.height);
154 						
155 		Vector3DUtil.scaleBy(obj1.normal, weightStart);
156 		Vector3DUtil.scaleBy(obj2.normal, weightEnd);
157 		out.normalOut = Vector3DUtil.add(obj1.normal, obj2.normal);
158 		Vector3DUtil.scaleBy(out.normalOut, 1 / (weightStart + weightEnd));
159 						
160 		out.fracOut = obj1.height / (obj1.height + depthEnd + JNumber3D.NUM_TINY);
161 		out.posOut = seg.getPoint(out.fracOut);
162 						
163 		return true;
164 	};

178 			n = max;
