 28 	 * @function getIntersectionLine

 32 	 * @type array
 33 	 **/
 34 	JMath3D.getIntersectionLine=function(v, v0, v1){
 35 		var d0 = v[0] * v0[0] + v[1] * v0[1] + v[2] * v0[2] - v[3];
 36 		var d1 = v[0] * v1[0] + v[1] * v1[1] + v[2] * v1[2] - v[3];
 37 		var m = d1 / (d1 - d0);
 38 		return [v1[0] + (v0[0] - v1[0]) * m,
 39 				v1[1] + (v0[1] - v1[1]) * m,
 40 				v1[2] + (v0[2] - v1[2]) * m, 
 41 				0];
 42 	};
