 27 	 **/

 68 	/**
 69 	 * @function distance determines the distance between vectors v1 and v2
 70 	 * @param v1 {array} in the format [x,y,z,w]
 71 	 * @param v2 {array} in the format [x,y,z,w]
 72 	 * @type number

 78 		var y=pow(v1[1]-v2[1], 2);

 81 	};

 84 	 * @function dotProduct determines the dot product for two vectors

 87 	 * @type number
 88 	 **/

100 		return [v1[1]*v2[2]-v1[2]*v2[1],v1[2]*v2[0]-v1[0]*v2[2],v1[0]*v2[1]-v1[1]*v2[0],0];
101 	};
