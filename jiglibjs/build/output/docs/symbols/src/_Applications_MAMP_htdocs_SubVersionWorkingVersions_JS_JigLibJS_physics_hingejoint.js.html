 21 (function(jigLib){
 22 	var Vector3DUtil=jigLib.Vector3DUtil;
 23 	var JMatrix3D=jigLib.JMatrix3D;
 24 	var JNumber3D=jigLib.JNumber3D;
 25 	var JConstraintMaxDistance=jigLib.JConstraintMaxDistance;
 26 	var JConstraintPoint=jigLib.JConstraintPoint;
 27 
 28 	/**
 29 	 * @author Muzer(muzerly@gmail.com)
 30 	 * 
 31 	 * @name HingeJoint
 32 	 * @class HingeJoint hinge connector for two rigid bodies
 33 	 * @extends PhysicsController
 34 	 * @requires Vector3DUtil
 35 	 * @requires JMatrix3D
 36 	 * @requires JNumber3D
 37 	 * @requires JConstraintMaxDistance
 38 	 * @requires JConstraintPoint
 39 	 * @constant {number} MAX_HINGE_ANGLE_LIMIT
 40 	 * @property {array} _hingeAxis
 41 	 * @property {array} _hingePosRel0
 42 	 * @property {RigidBody} body0 the first rigid body 
 43 	 * @property {RigidBody} body1 the second rigid body 
 44 	 * @property {boolean} _usingLimit
 45 	 * @property {boolean} _hingeEnabled
 46 	 * @property {boolean} _broken
 47 	 * @property {number} _damping
 48 	 * @property {number} _extraTorque
 49 	 * @property {array} sidePointConstraints used to store 2 JConstraintMaxDistance instances
 50 	 * @property {JConstraintPoint} midPointConstraint
 51 	 * @property {JConstraintMaxDistance} maxDistanceConstraint
 52 	 * @property {array} r a 3D vector
 53 	 * @constructor
 54 	 * @param {RigidBody} _body0 the first body of the constrained pair
 55 	 * @param {RigidBody} _body1 the second body of the constrained pair
 56 	 * @param {array} _hingeAxis
 57 	 * @param {array} _hingePosRel0
 58 	 * @param {number} hingeHalfWidth
 59 	 * @param {number} hingeFwdAngle
 60 	 * @param {number} hingeBckAngle
 61 	 * @param {number} sidewaysSlack
 62 	 * @param {number} damping
 63 	 **/
 64 	var HingeJoint=function(body0, body1, hingeAxis, hingePosRel0, hingeHalfWidth, hingeFwdAngle, hingeBckAngle, sidewaysSlack, damping){
 65 		this._body0 = body0;
 66 		this._body1 = body1;
 67 		this._hingeAxis = hingeAxis.slice(0);
 68 		this._hingePosRel0 = hingePosRel0.slice(0);
 69 		this._usingLimit = false;
 70 		this._hingeEnabled = false;
 71 		this._broken = false;
 72 		this._damping = damping;
 73 		this._extraTorque = 0;
 74 
 75 		Vector3DUtil.normalize(this._hingeAxis);
 76 		var _hingePosRel1 = Vector3DUtil.add(this._body0.get_currentState().position, Vector3DUtil.subtract(this._hingePosRel0, this._body1.get_currentState().position));
 77 
 78 		var relPos0a = Vector3DUtil.add(this._hingePosRel0, JNumber3D.getScaleVector(this._hingeAxis, hingeHalfWidth));
 79 		var relPos0b = Vector3DUtil.subtract(this._hingePosRel0, JNumber3D.getScaleVector(this._hingeAxis, hingeHalfWidth));
 80 
 81 		var relPos1a = Vector3DUtil.add(_hingePosRel1, JNumber3D.getScaleVector(this._hingeAxis, hingeHalfWidth));
 82 		var relPos1b = Vector3DUtil.subtract(_hingePosRel1, JNumber3D.getScaleVector(this._hingeAxis, hingeHalfWidth));
 83 
 84 		var timescale = 1 / 20;
 85 		var allowedDistanceMid = 0.005;
 86 		var allowedDistanceSide = sidewaysSlack * hingeHalfWidth;
 87 
 88 		this.sidePointConstraints = [];
 89 		this.sidePointConstraints[0] = new JConstraintMaxDistance(this._body0, relPos0a, this._body1, relPos1a, allowedDistanceSide);
 90 		this.sidePointConstraints[1] = new JConstraintMaxDistance(this._body0, relPos0b, this._body1, relPos1b, allowedDistanceSide);
 91 
 92 		this.midPointConstraint = new JConstraintPoint(this._body0, this._hingePosRel0, this._body1, _hingePosRel1, allowedDistanceMid, timescale);
 93 
 94 		if (hingeFwdAngle <= this.MAX_HINGE_ANGLE_LIMIT){
 95 			var perpDir = Vector3DUtil.Y_AXIS;
 96 			if (Vector3DUtil.dotProduct(perpDir, this._hingeAxis) > 0.1){
 97 				perpDir[0] = 1;
 98 				perpDir[1] = 0;
 99 				perpDir[2] = 0;
100 			}
101 			var sideAxis = Vector3DUtil.crossProduct(this._hingeAxis, perpDir);
102 			perpDir = Vector3DUtil.crossProduct(sideAxis, this._hingeAxis);
103 			Vector3DUtil.normalize(perpDir);
104 
105 			var len = 10 * hingeHalfWidth;
106 			var hingeRelAnchorPos0 = JNumber3D.getScaleVector(perpDir, len);
107 			var angleToMiddle = 0.5 * (hingeFwdAngle - hingeBckAngle);
108 			var hingeRelAnchorPos1 = hingeRelAnchorPos0.slice(0);
109 			JMatrix3D.multiplyVector(JMatrix3D.getRotationMatrix(this._hingeAxis[0], this._hingeAxis[1], this._hingeAxis[2], -angleToMiddle), hingeRelAnchorPos1);
110 
111 			var hingeHalfAngle = 0.5 * (hingeFwdAngle + hingeBckAngle);
112 			var allowedDistance = len * 2 * Math.sin(0.5 * hingeHalfAngle * Math.PI / 180);
113 
114 			var hingePos = Vector3DUtil.add(this._body1.get_currentState().position, this._hingePosRel0);
115 			var relPos0c = Vector3DUtil.add(hingePos, Vector3DUtil.subtract(hingeRelAnchorPos0, this._body0.get_currentState().position));
116 			var relPos1c = Vector3DUtil.add(hingePos, Vector3DUtil.subtract(hingeRelAnchorPos1, this._body1.get_currentState().position));
117 
118 			this.maxDistanceConstraint = new JConstraintMaxDistance(this._body0, relPos0c, this._body1, relPos1c, allowedDistance);
119 			this._usingLimit = true;
120 		}
121 		if (this._damping <= 0){
122 			this._damping = -1;
123 		}else{
124 			this._damping = JNumber3D.getLimiteNumber(this._damping, 0, 1);
125 		}
126 
127 		this.enableHinge();
128 	};
129 	jigLib.extend(HingeJoint, jigLib.PhysicsController);
130 	
131 	HingeJoint.prototype.MAX_HINGE_ANGLE_LIMIT = 150;
132 	HingeJoint.prototype._hingeAxis = null;
133 	HingeJoint.prototype._hingePosRel0 = null;
134 	HingeJoint.prototype._body0 = null;
135 	HingeJoint.prototype._body1 = null;
136 	HingeJoint.prototype._usingLimit = null;
137 	HingeJoint.prototype._hingeEnabled = null;
138 	HingeJoint.prototype._broken = null;
139 	HingeJoint.prototype._damping = null;
140 	HingeJoint.prototype._extraTorque = null;
141 	
142 	HingeJoint.prototype.sidePointConstraints = null;
143 	HingeJoint.prototype.midPointConstraint = null;
144 	HingeJoint.prototype.maxDistanceConstraint = null;
145 
146 	/**
147 	 * @function enableHinge enable the joint
148 	 * @type void
149 	 **/
150 	HingeJoint.prototype.enableHinge=function(){
151 		if (this._hingeEnabled) return;
152 		
153 		this.midPointConstraint.enableConstraint();
154 		this.sidePointConstraints[0].enableConstraint();
155 		this.sidePointConstraints[1].enableConstraint();
156 		if (this._usingLimit && !this._broken)
157 			this.maxDistanceConstraint.enableConstraint();
158 
159 		this.enableController();
160 		this._hingeEnabled = true;
161 	};
162 
163 	/**
164 	 * @function disableHinge disable the joint
165 	 * @type void
166 	 **/
167 	HingeJoint.prototype.disableHinge=function(){
168 		if (!this._hingeEnabled) return;
169 
170 		this.midPointConstraint.disableConstraint();
171 		this.sidePointConstraints[0].disableConstraint();
172 		this.sidePointConstraints[1].disableConstraint();
173 
174 		if (this._usingLimit && !this._broken)
175 			this.maxDistanceConstraint.disableConstraint();
176 
177 		this.disableController();
178 		this._hingeEnabled = false;
179 	};
180 
181 	/**
182 	 * @function breakHinge break the joint
183 	 * @type void
184 	 **/
185 	HingeJoint.prototype.breakHinge=function(){
186 		if (this._broken) return;
187 
188 		if (this._usingLimit)
189 			this.maxDistanceConstraint.disableConstraint();
190 
191 		this._broken = true;
192 	};
193 
194 	/**
195 	 * @function mendHinge repair the joint
196 	 * @type void
197 	 **/
198 	HingeJoint.prototype.mendHinge=function(){
199 		if (!this._broken)
200 			return;
201 
202 		if (this._usingLimit)
203 			this.maxDistanceConstraint.enableConstraint();
204 
205 		this._broken = false;
206 	};
207 
208 	/**
209 	 * @function setExtraTorque setter for _extraTorque
210 	 * @param {number} torque
211 	 * @type void
212 	 **/
213 	HingeJoint.prototype.setExtraTorque=function(torque){
214 		this._extraTorque = torque;
215 	};
216 
217 	/**
218 	 * @function getExtraTorque getter for _extraTorque
219 	 * @type number
220 	 **/
221 	HingeJoint.prototype.getHingeEnabled=function(){
222 		return this._hingeEnabled;
223 	};
224 
225 	/**
226 	 * @function isBroken getter for _broken
227 	 * @type boolean
228 	 **/
229 	HingeJoint.prototype.isBroken=function(){
230 		return this._broken;
231 	};
232 
233 	/**
234 	 * @function getHingePosRel0 getter for _hingePosRel0
235 	 * @type array
236 	 **/
237 	HingeJoint.prototype.getHingePosRel0=function(){
238 		return this._hingePosRel0;
239 	};
240 
241 	/**
242 	 * @function updateController updates this physics controller
243 	 * @see PhysicsSystem.updateController
244 	 * @param {number} dt a UNIX timestamp
245 	 * @type void
246 	 **/
247 	HingeJoint.prototype.updateController=function(dt){
248 		if (this._damping > 0){
249 			var hingeAxis = Vector3DUtil.subtract(this._body1.get_currentState().rotVelocity, this._body0.get_currentState().rotVelocity);
250 			Vector3DUtil.normalize(hingeAxis);
251 
252 			var angRot1 = Vector3DUtil.dotProduct(this._body0.get_currentState().rotVelocity, hingeAxis);
253 			var angRot2 = Vector3DUtil.dotProduct(this._body1.get_currentState().rotVelocity, hingeAxis);
254 
255 			var avAngRot = 0.5 * (angRot1 + angRot2);
256 			var frac = 1 - this._damping;
257 			var newAngRot1= avAngRot + (angRot1 - avAngRot) * frac;
258 			var newAngRot2= avAngRot + (angRot2 - avAngRot) * frac;
259 
260 			var newAngVel1 = Vector3DUtil.add(this._body0.get_currentState().rotVelocity, JNumber3D.getScaleVector(hingeAxis, newAngRot1 - angRot1));
261 			var newAngVel2 = Vector3DUtil.add(this._body1.get_currentState().rotVelocity, JNumber3D.getScaleVector(hingeAxis, newAngRot2 - angRot2));
262 
263 			this._body0.setAngVel(newAngVel1);
264 			this._body1.setAngVel(newAngVel2);
265 		}
266 
267 		if (this._extraTorque != 0){
268 			var torque1 = this._hingeAxis.slice(0);
269 			JMatrix3D.multiplyVector(this._body0.get_currentState().get_orientation(), torque1);
270 			torque1 = JNumber3D.getScaleVector(torque1, this._extraTorque);
271 
272 			this._body0.addWorldTorque(torque1);
273 			this._body1.addWorldTorque(JNumber3D.getScaleVector(torque1, -1));
274 		}
275 	};
276 	
277 	jigLib.HingeJoint=HingeJoint;
