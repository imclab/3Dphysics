318 	 * @function setSolverType select which solver type should be used for the simulation
319 	 * @see JConfig.solverType
320 	 * @param {string} type
321 	 * @type void
322 	 **/
323 	PhysicsSystem.prototype.setSolverType=function(type){
324 		switch (type)
325 		{
326 			case "FAST":
327 				this.preProcessCollisionFn = this.preProcessCollisionFast;
328 				this.preProcessContactFn = this.preProcessCollisionFast;
329 				this.processCollisionFn = this.processCollision;
330 				this.processContactFn = this.processCollision;
331 				return;
332 			case "NORMAL":
333 				this.preProcessCollisionFn = this.preProcessCollisionNormal;
334 				this.preProcessContactFn = this.preProcessCollisionNormal;
335 				this.processCollisionFn = this.processCollision;
336 				this.processContactFn = this.processCollision;
337 				return;
338 			case "ACCUMULATED":
339 				this.preProcessCollisionFn = this.preProcessCollisionAccumulated;
340 				this.preProcessContactFn = this.preProcessCollisionAccumulated;
341 				this.processCollisionFn = this.processCollision;
342 				this.processContactFn = this.processCollisionAccumulated;
343 				return;
344 			default:
345 				this.preProcessCollisionFn = this.preProcessCollisionNormal;
346 				this.preProcessContactFn = this.preProcessCollisionNormal;
347 				this.processCollisionFn = this.processCollision;
348 				this.processContactFn = this.processCollision;
349 				return;
350 		}
351 	};

401 	/**
402 	 * @function preProcessCollisionFast a fast-but-inaccurate pre-processor
403 	 * @param {CollisionInfo} collision
404 	 * @param {number} dt a UNIX timestamp
405 	 * @type void
406 	 **/
407 	PhysicsSystem.prototype.preProcessCollisionFast=function(collision, dt){
408 		collision.satisfied=false;
409 
410 		var body0 = collision.objInfo.body0;
411 		var body1 = collision.objInfo.body1;
412 
413 		var N = collision.dirToBody;
414 		var timescale = JConfig.numPenetrationRelaxationTimesteps * dt;
415 		var approachScale = 0;
416 		var ptInfo;
417 		var tempV;
418 		var ptNum = collision.pointInfo.length;
419 		var numTiny = JNumber3D.NUM_TINY;
420 
421 		if (ptNum > 1){
422 			var avR0 = [0,0,0,0];
423 			var avR1 = [0,0,0,0];
424 			var avDepth = 0;
425 
426 			for (var i = 0; i < ptNum; i++){
427 				ptInfo = collision.pointInfo[i];
428 				avR0 = Vector3DUtil.add(avR0, ptInfo.r0);
429 				avR1 = Vector3DUtil.add(avR1, ptInfo.r1);
430 				avDepth += ptInfo.initialPenetration;
431 			}
432 			avR0 = JNumber3D.getDivideVector(avR0, ptNum);
433 			avR1 = JNumber3D.getDivideVector(avR1, ptNum);
434 			avDepth /= ptNum;
435 
436 			var colPI = new CollPointInfo();
437 			colPI.r0 = avR0;
438 			colPI.r1 = avR1;
439 			colPI.initialPenetration = avDepth;
440 			collision.pointInfo = [colPI];
441 		}
442 		
443 		// removed loop because collision.pointInfo.length can only ever be 1 - Jim Sangwine
444 		ptInfo = collision.pointInfo[0];
445 		if (!body0.get_movable()){
446 			ptInfo.denominator = 0;
447 		}else{
448 			tempV = Vector3DUtil.crossProduct(ptInfo.r0, N);
449 			JMatrix3D.multiplyVector(body0.get_worldInvInertia(), tempV);
450 			ptInfo.denominator = body0.get_invMass() + Vector3DUtil.dotProduct(N, Vector3DUtil.crossProduct(tempV, ptInfo.r0));
451 		}
452 		if (body1.get_movable()){
453 			tempV = Vector3DUtil.crossProduct(ptInfo.r1, N);
454 			JMatrix3D.multiplyVector(body1.get_worldInvInertia(), tempV);
455 			ptInfo.denominator += (body1.get_invMass() + Vector3DUtil.dotProduct(N, Vector3DUtil.crossProduct(tempV, ptInfo.r1)));
456 		}
457 		if (ptInfo.denominator < numTiny)
458 			ptInfo.denominator = numTiny;
459 
460 		if (ptInfo.initialPenetration > JConfig.allowedPenetration){
461 			ptInfo.minSeparationVel = (ptInfo.initialPenetration - JConfig.allowedPenetration) / timescale;
462 		}else{
463 			approachScale = -0.1 * (ptInfo.initialPenetration - JConfig.allowedPenetration) / JConfig.allowedPenetration;
464 			if (approachScale < numTiny)
465 				approachScale = numTiny;
466 			else if (approachScale > 1)
467 				approachScale = 1;
468 			var max = (dt > numTiny) ? dt : numTiny; // ~7x quicker than Math.max in Chromium, ~4x quicker in WebKit and marginally slower in Minefield
469 			ptInfo.minSeparationVel = approachScale * (ptInfo.initialPenetration - JConfig.allowedPenetration) / max;
470 		}
471 		
472 		if (ptInfo.minSeparationVel > this._maxVelMag)
473 			ptInfo.minSeparationVel = this._maxVelMag;
474 	};

476 	/**
477 	 * @function preProcessCollisionNormal a special pre-processor for the normal solver
478 	 * @param {CollisionInfo} collision
479 	 * @param {number} dt a UNIX timestamp
480 	 * @type void
481 	 **/
482 	PhysicsSystem.prototype.preProcessCollisionNormal=function(collision, dt){
483 		collision.satisfied = false;
484 
485 		var body0 = collision.objInfo.body0;
486 		var body1 = collision.objInfo.body1;
487 
488 		var N = collision.dirToBody;
489 		var timescale= JConfig.numPenetrationRelaxationTimesteps * dt;
490 		var approachScale = 0;
491 		var ptInfo;
492 		var tempV;
493 		var len= collision.pointInfo.length;
494 		for (var i = 0; i < len; i++){
495 			ptInfo = collision.pointInfo[i];
496 			if (!body0.get_movable()){
497 				ptInfo.denominator = 0;
498 			}else{
499 				tempV = Vector3DUtil.crossProduct(ptInfo.r0, N);
500 				JMatrix3D.multiplyVector(body0.get_worldInvInertia(), tempV);
501 				ptInfo.denominator = body0.get_invMass() + Vector3DUtil.dotProduct(N, Vector3DUtil.crossProduct(tempV, ptInfo.r0));
502 			}
503 
504 			if (body1.get_movable()){
505 				tempV = Vector3DUtil.crossProduct(ptInfo.r1, N);
506 				JMatrix3D.multiplyVector(body1.get_worldInvInertia(), tempV);
507 				ptInfo.denominator += (body1.get_invMass() + Vector3DUtil.dotProduct(N, Vector3DUtil.crossProduct(tempV, ptInfo.r1)));
508 			}
509 
510 			if (ptInfo.denominator < JNumber3D.NUM_TINY)
511 				ptInfo.denominator = JNumber3D.NUM_TINY;
512 
513 			if (ptInfo.initialPenetration > JConfig.allowedPenetration){
514 				ptInfo.minSeparationVel = (ptInfo.initialPenetration - JConfig.allowedPenetration) / timescale;
515 			}else{
516 				approachScale = -0.1 * (ptInfo.initialPenetration - JConfig.allowedPenetration) / JConfig.allowedPenetration;
517 				if (approachScale < JNumber3D.NUM_TINY)
518 					approachScale = JNumber3D.NUM_TINY;
519 				else if (approachScale > 1)
520 					approachScale = 1;
521 				
522 				var max=(dt > JNumber3D.NUM_TINY) ? dt : JNumber3D.NUM_TINY;
523 				ptInfo.minSeparationVel = approachScale * (ptInfo.initialPenetration - JConfig.allowedPenetration) / max;
524 			}
525 			if (ptInfo.minSeparationVel > this._maxVelMag)
526 				ptInfo.minSeparationVel = this._maxVelMag;
527 		}
528 	};

530 	/**
531 	 * @function preProcessCollisionAccumulated a special pre-processor for the accumulated solver
532 	 * @param {CollisionInfo} collision
533 	 * @param {number} dt a UNIX timestamp
534 	 * @type void
535 	 **/
536 	PhysicsSystem.prototype.preProcessCollisionAccumulated=function(collision, dt){
537 		collision.satisfied = false;
538 		var body0 = collision.objInfo.body0;
539 		var body1 = collision.objInfo.body1;
540 		var N = collision.dirToBody;
541 		var timescale = JConfig.numPenetrationRelaxationTimesteps * dt;
542 		var tempV;
543 		var ptInfo;
544 		var initMinAllowedPen;
545 		var approachScale = 0;
546 		var numTiny = JNumber3D.NUM_TINY;
547 		var allowedPenetration = JConfig.allowedPenetration;
548 		var len = collision.pointInfo.length;
549 		for (var i = 0; i < len; i++){
550 			ptInfo = collision.pointInfo[i];
551 			initMinAllowedPen = ptInfo.initialPenetration - allowedPenetration;
552 			if (!body0.get_movable()){
553 				ptInfo.denominator = 0;
554 			}else{
555 				tempV = Vector3DUtil.crossProduct(ptInfo.r0, N);
556 				JMatrix3D.multiplyVector(body0.get_worldInvInertia(), tempV);
557 				ptInfo.denominator = body0.get_invMass() + Vector3DUtil.dotProduct(N, Vector3DUtil.crossProduct(tempV, ptInfo.r0));
558 			}
559 
560 			if (body1.get_movable()){
561 				tempV = Vector3DUtil.crossProduct(ptInfo.r1, N);
562 				JMatrix3D.multiplyVector(body1.get_worldInvInertia(), tempV);
563 				ptInfo.denominator += (body1.get_invMass() + Vector3DUtil.dotProduct(N, Vector3DUtil.crossProduct(tempV, ptInfo.r1)));
564 			}
565 			if (ptInfo.denominator < numTiny) ptInfo.denominator = numTiny;
566 
567 			if (ptInfo.initialPenetration > allowedPenetration){
568 				ptInfo.minSeparationVel = initMinAllowedPen / timescale;
569 			}else{
570 				approachScale = -0.1 * initMinAllowedPen / allowedPenetration;
571 				
572 				if (approachScale < numTiny) approachScale = numTiny;
573 				else if (approachScale > 1) approachScale = 1;
574 				
575 				var max=(dt>numTiny) ? dt : numTiny;
576 				ptInfo.minSeparationVel = approachScale * initMinAllowedPen / max;
577 			}
578 
579 			ptInfo.accumulatedNormalImpulse = 0;
580 			ptInfo.accumulatedNormalImpulseAux = 0;
581 			ptInfo.accumulatedFrictionImpulse = [0,0,0,0];
582 
583 			var bestDistSq = 0.04;
584 			var bp = new BodyPair(body0, body1, [0,0,0,0], [0,0,0,0]);
585 
586 			for(var j=0, ccl=this._cachedContacts.length; j<ccl; j++){
587 				var cont=this._cachedContacts[j];
588 				var cpair=cont.pair;
589 
590 				if (bp.body0 != cpair.body0 || bp.body1 == cpair.body1)
591 					continue;
592 
593 				var distSq = (cpair.body0 == body0)  ? Vector3DUtil.get_lengthSquared(Vector3DUtil.subtract(cpair.r, ptInfo.r0)) 
594 													 : Vector3DUtil.get_lengthSquared(Vector3DUtil.subtract(cpair.r, ptInfo.r1));
595 
596 				if (distSq < bestDistSq){
597 					bestDistSq = distSq;
598 					ptInfo.accumulatedNormalImpulse = this._cachedContacts[j].impulse.normalImpulse;
599 					ptInfo.accumulatedNormalImpulseAux = this._cachedContacts[j].impulse.normalImpulseAux;
600 					ptInfo.accumulatedFrictionImpulse = this._cachedContacts[j].impulse.frictionImpulse;
601 					if (this._cachedContacts[j].pair.body0 != body0){
602 						ptInfo.accumulatedFrictionImpulse = JNumber3D.getScaleVector(ptInfo.accumulatedFrictionImpulse, -1);
603 					}
604 				}
605 			}
606 			
607 			var impulse;
608 			if (ptInfo.accumulatedNormalImpulse != 0){
609 				impulse = JNumber3D.getScaleVector(N, ptInfo.accumulatedNormalImpulse);
610 				impulse = Vector3DUtil.add(impulse, ptInfo.accumulatedFrictionImpulse);
611 				body0.applyBodyWorldImpulse(impulse, ptInfo.r0);
612 				body1.applyBodyWorldImpulse(JNumber3D.getScaleVector(impulse, -1), ptInfo.r1);
613 			}
614 			if (ptInfo.accumulatedNormalImpulseAux != 0){
615 				impulse = JNumber3D.getScaleVector(N, ptInfo.accumulatedNormalImpulseAux);
616 				body0.applyBodyWorldImpulseAux(impulse, ptInfo.r0);
617 				body1.applyBodyWorldImpulseAux(JNumber3D.getScaleVector(impulse, -1), ptInfo.r1);
618 			}
619 		}

622 	/**
623 	 * @function processCollision handle an individual collision by classifying it, calculating
624 	 * impulse, applying impulse and updating the velocities of the objects. Allows over-riding of the elasticity.
625 	 * 
626 	 * @param {CollisionInfo} collision
627 	 * @param {number} dt a UNIX timestamp
628 	 * @returns true if an impulse was applied, otherwise false
629 	 * @type boolean
630 	 **/
631 	PhysicsSystem.prototype.processCollision=function(collision, dt){
632 		collision.satisfied = true;
633 
634 		var body0 = collision.objInfo.body0;
635 		var body1 = collision.objInfo.body1;
636 
637 		var gotOne = false;
638 		var N = collision.dirToBody;
639 
640 		var deltaVel = 0;
641 		var normalVel = 0;
642 		var finalNormalVel = 0;
643 		var normalImpulse= 0;
644 		var impulse;
645 		var Vr0;
646 		var Vr1;
647 		var ptInfo;
648 		
649 		// tracking var for the collision event
650 		var appliedImpulse = [0,0,0,0];
651 		
652 		var len = collision.pointInfo.length;
653 		for (var i = 0; i < len; i++){
654 			ptInfo = collision.pointInfo[i];
655 
656 			Vr0 = body0.getVelocity(ptInfo.r0);
657 			Vr1 = body1.getVelocity(ptInfo.r1);
658 
659 			normalVel = Vector3DUtil.dotProduct(Vector3DUtil.subtract(Vr0, Vr1), N);
660 			if (normalVel > ptInfo.minSeparationVel)
661 				continue;
662 
663 			finalNormalVel = -1 * collision.mat.get_restitution() * normalVel;
664 			if (finalNormalVel < this._minVelForProcessing)
665 				finalNormalVel = ptInfo.minSeparationVel;
666 
667 			deltaVel = finalNormalVel - normalVel;
668 			if (deltaVel <= this._minVelForProcessing)
669 				continue;
670 
671 			normalImpulse = deltaVel / ptInfo.denominator;
672 
673 			impulse = JNumber3D.getScaleVector(N, normalImpulse);
674 			
675 			// Performance improvement? Why continue if impulse is nil?
676 			if (Vector3DUtil.getSum(impulse) > 0)
677 			{
678 				appliedImpulse = Vector3DUtil.add(appliedImpulse, impulse); // keep track of the total impulse applied
679 				body0.applyBodyWorldImpulse(impulse, ptInfo.r0);
680 				body1.applyBodyWorldImpulse(JNumber3D.getScaleVector(impulse, -1), ptInfo.r1);
681 				gotOne = true;
682 			}
683 
684 			var tempV;
685 			var VR = Vr0.slice(0);
686 			if(body1.get_movable()) VR = Vector3DUtil.subtract(Vr0, Vr1);
687 			var tangent_vel = Vector3DUtil.subtract(VR, JNumber3D.getScaleVector(N, Vector3DUtil.dotProduct(VR, N)));
688 			var tangent_speed = Vector3DUtil.get_length(tangent_vel);
689 
690 			if (tangent_speed > this._minVelForProcessing){
691 				var T = JNumber3D.getDivideVector(tangent_vel, -tangent_speed);
692 				var denominator = 0;
693 
694 				if (body0.get_movable()){
695 					tempV = Vector3DUtil.crossProduct(ptInfo.r0, T);
696 					JMatrix3D.multiplyVector(body0.get_worldInvInertia(), tempV);
697 					denominator = body0.get_invMass() + Vector3DUtil.dotProduct(T, Vector3DUtil.crossProduct(tempV, ptInfo.r0));
698 				}
699 
700 				if (body1.get_movable()){
701 					tempV = Vector3DUtil.crossProduct(ptInfo.r1, T);
702 					JMatrix3D.multiplyVector(body1.get_worldInvInertia(), tempV);
703 					denominator += (body1.get_invMass() + Vector3DUtil.dotProduct(T, Vector3DUtil.crossProduct(tempV, ptInfo.r1)));
704 				}
705 
706 				if (denominator > JNumber3D.NUM_TINY){
707 					var impulseToReverse = tangent_speed / denominator;
708 
709 					T = JNumber3D.getScaleVector(T, impulseToReverse);
710 					body0.applyBodyWorldImpulse(T, ptInfo.r0);
711 					body1.applyBodyWorldImpulse(JNumber3D.getScaleVector(T, -1), ptInfo.r1);
712 				}
713 			}
714 		}
715 
716 		if (gotOne){
717 			body0.setConstraintsAndCollisionsUnsatisfied();
718 			body1.setConstraintsAndCollisionsUnsatisfied();
719 			// dispatch collision events
720 			body0.dispatchCollisionEvent(body1, appliedImpulse);
721 			body1.dispatchCollisionEvent(body0, JNumber3D.getScaleVector(appliedImpulse, -1));
722 		}
723 		return gotOne;
724 	};

829 						denominator += (body1.invMass + Vector3DUtil.dotProduct(T, Vector3DUtil.crossProduct(tempV, ptInfo.r1)));
830 					}
831 					if (denominator > JNumber3D.NUM_TINY){
832 						var impulseToReverse = tangent_speed / denominator;

931 		var info;
