109 		this._oldState = new PhysicsState();

111 		this._invOrientation = JMatrix3D.getInverseMatrix(this._currState.get_orientation());

119 		this._rotVelDamping = [0.5, 0.5, 0.5, 0];
120 		this._maxLinVelocities = 500;
121 		this._maxRotVelocities = 50;
122 
123 		this._velChanged = false;
124 		this._inactiveTime = 0;
125 		
126 		this._doShockProcessing = true;
127 
128 		this.isActive = this._activity = true;
129 		this._movable = true;
130 		this._origMovable = true;
131 
132 		this.collisions = [];
133 		this._constraints = [];
134 		this._nonCollidables = [];

137 		this._bodiesToBeActivatedOnMovement = [];
138 		this._lastPositionForDeactivation = this._currState.position.slice(0);
139 		this._lastOrientationForDeactivation = this._currState.get_orientation().clone();

142 		this._boundingSphere = 0;
143 		this._boundingBox = new JAABox([0,0,0,0], [0,0,0,0]);

205 	/**
206 	 * @function dispatchCollisionEvent dispatches a JCollisionEvent
207 	 * @param {RigidBody} body the other body involved in the collision
208 	 * @param {Array} impulse a 3D vector representing the impulse applied to this body as a result of the collision
209 	 */
210 	RigidBody.prototype.dispatchCollisionEvent=function(body, impulse)
211 	{
212 		if (Vector3DUtil.getSum(impulse) < this.minImpulseForCollisionEvent)
213 			return;
214 		
215 		this.dispatchEvent(new JCollisionEvent(body, impulse));
216 	};

338 		matrix3D.appendRotation(this._rotationX, Vector3DUtil.X_AXIS);
339 		matrix3D.appendRotation(this._rotationY, Vector3DUtil.Y_AXIS);

522 	 * @function addWorldForce add forces in the world coordinate frame
523 	 * @param {array} f force expressed as a 3D vector
524 	 * @param {array} p position of origin of the force expressed as a 3D vector 
525 	 * @type void
526 	 **/
527 	RigidBody.prototype.addWorldForce=function(f, p){
528 		if (!this._movable) return;
529 		
530 		this._force = Vector3DUtil.add(this._force, f);
531 		this.addWorldTorque(Vector3DUtil.crossProduct(Vector3DUtil.subtract(p, this._currState.position), f));
532 		this._velChanged = true;
533 		this.setActive();
534 	};

548 		this.addWorldForce(f, Vector3DUtil.add(this._currState.position, p));

560 	/**
561 	 * @function applyWorldImpulse add impulses in the world coordinate frame
562 	 * @param {array} impulse impulse expressed as a 3D vector
563 	 * @param {array} pos position of origin of the impulse expressed as a 3D vector 
564 	 * @type void
565 	 **/
566 	RigidBody.prototype.applyWorldImpulse=function(impulse, pos){
567 		if (!this._movable) return;
568 		
569 		this._currState.linVelocity = Vector3DUtil.add(this._currState.linVelocity, JNumber3D.getScaleVector(impulse, this._invMass));
570 
571 		var rotImpulse = Vector3DUtil.crossProduct(Vector3DUtil.subtract(pos, this._currState.position), impulse);
572 		JMatrix3D.multiplyVector(this._worldInvInertia, rotImpulse);
573 		this._currState.rotVelocity = Vector3DUtil.add(this._currState.rotVelocity, rotImpulse);
574 
575 		this._velChanged = true;
576 	};

579 	 * @function applyWorldImpulseAux

583 	 **/
584 	RigidBody.prototype.applyWorldImpulseAux=function(impulse, pos){
585 		if (!this._movable) return;

588

599 	 * @param {array} delta impulse delta expressed as a 3D vector 
600 	 * @type void
601 	 **/
602 	RigidBody.prototype.applyBodyWorldImpulse=function(impulse, delta){
603 		if (!this._movable) return;
604 		
605 		this._currState.linVelocity = Vector3DUtil.add(this._currState.linVelocity, JNumber3D.getScaleVector(impulse, this._invMass));
606 		var rotImpulse = Vector3DUtil.crossProduct(delta, impulse);
607 		JMatrix3D.multiplyVector(this._worldInvInertia, rotImpulse);
608 		this._currState.rotVelocity = Vector3DUtil.add(this._currState.rotVelocity, rotImpulse);
609 
610 		this._velChanged = true;

692 	 * @function updateVelocity update the position with the auxiliary velocities, and zeros them
693 	 * @param {number} dt a UNIX timestamp
694 	 * @type void
695 	 **/
696 	RigidBody.prototype.updatePositionWithAux=function(dt){
697 		if (!this._movable || !this._activity){
698 			this._currLinVelocityAux = [0,0,0,0];
699 			this._currRotVelocityAux = [0,0,0,0];
700 			return;
701 		}
702 		
703 		var ga = jigLib.PhysicsSystem.getInstance().get_gravityAxis();
704 		
705 		if (ga != -1){
706 			var arr = this._currLinVelocityAux.slice(0);
707 			arr[(ga + 1) % 3] *= 0.1;
708 			arr[(ga + 2) % 3] *= 0.1;
709 			JNumber3D.copyFromArray(this._currLinVelocityAux, arr);
710 		}
711 
712 		var angMomBefore = this._currState.rotVelocity.slice(0);
713 		JMatrix3D.multiplyVector(this._worldInertia, angMomBefore);
714 		
715 		this._currState.position = Vector3DUtil.add(this._currState.position, JNumber3D.getScaleVector(Vector3DUtil.add(this._currState.linVelocity, this._currLinVelocityAux), dt));
716 
717 		var dir = Vector3DUtil.add(this._currState.rotVelocity, this._currRotVelocityAux);
718 		var ang = Vector3DUtil.get_length(dir) * 180 / Math.PI;
719 		if (ang > 0){
720 			Vector3DUtil.normalize(dir);
721 			ang *= dt;
722 			var rot = JMatrix3D.getRotationMatrix(dir[0], dir[1], dir[2], ang);
723 			this._currState.set_orientation(JMatrix3D.getAppendMatrix3D(this._currState.get_orientation(), rot));
724 				
725 			this.updateInertia();
726 		}
727 		this._currLinVelocityAux = [0,0,0,0];
728 		this._currRotVelocityAux = [0,0,0,0];
729 		
730 		JMatrix3D.multiplyVector(this._worldInvInertia, angMomBefore);
731 		this._currState.rotVelocity = angMomBefore.slice(0);
732 			
733 		this.updateBoundingBox();
734 	};

744 	 * @function tryToFreeze provided for the use of Physics system
745 	 * @param {number} dt a UNIX timestamp
746 	 * @type void
747 	 **/
748 	RigidBody.prototype.tryToFreeze=function(dt){
749 		if (!this._movable || !this._activity)
750 			return;
751 		
752 		if (Vector3DUtil.get_length(Vector3DUtil.subtract(this._currState.position, this._lastPositionForDeactivation)) > JConfig.posThreshold){
753 			this._lastPositionForDeactivation = this._currState.position.slice(0);
754 			this._inactiveTime = 0;
755 			return;
756 		}
757 		
758 		var ot = JConfig.orientThreshold;
759 		var deltaMat = JMatrix3D.getSubMatrix(this._currState.get_orientation(), this._lastOrientationForDeactivation);
760 
761 		var cols = JMatrix3D.getCols(deltaMat);
762 
763 		if (Vector3DUtil.get_length(cols[0]) > ot || Vector3DUtil.get_length(cols[1]) > ot || Vector3DUtil.get_length(cols[2]) > ot){
764 			this._lastOrientationForDeactivation = this._currState.get_orientation().clone();
765 			this._inactiveTime = 0;
766 			return;
767 		}
768 
769 		if (this.getShouldBeActive()){
770 			return;
771 		}
772 
773 		this._inactiveTime += dt;
774 		if (this._inactiveTime > JConfig.deactivationTime){
775 			this._lastPositionForDeactivation = this._currState.position.slice(0);
776 			this._lastOrientationForDeactivation = this._currState.get_orientation().clone();
777 			this.setInactive();
778 		}
779 	};

940 	 * @function dampForDeactivation damp movement as the body approaches deactivation
941 	 * @type void
942 	 **/
943 	RigidBody.prototype.dampForDeactivation=function(){
944 		this._currState.linVelocity[0] *= this._linVelDamping[0];
945 		this._currState.linVelocity[1] *= this._linVelDamping[1];
946 		this._currState.linVelocity[2] *= this._linVelDamping[2];
947 		this._currState.rotVelocity[0] *= this._rotVelDamping[0];
948 		this._currState.rotVelocity[1] *= this._rotVelDamping[1];
949 		this._currState.rotVelocity[2] *= this._rotVelDamping[2];
950 			
951 		this._currLinVelocityAux[0] *= this._linVelDamping[0];
952 		this._currLinVelocityAux[1] *= this._linVelDamping[1];
953 		this._currLinVelocityAux[2] *= this._linVelDamping[2];
954 		this._currRotVelocityAux[0] *= this._rotVelDamping[0];
955 		this._currRotVelocityAux[1] *= this._rotVelDamping[1];
956 		this._currRotVelocityAux[2] *= this._rotVelDamping[2];
957 			
958 		var r = 0.5;
959 		var frac = this._inactiveTime / JConfig.deactivationTime;
960 		if (frac < r){
961 			return;
962 		}
963 
964 		var scale = 1 - ((frac - r) / (1 - r));
965 		if (scale < 0){
966 			scale = 0;
967 		}else if (scale > 1){
968 			scale = 1;
969 		}
970 		this._currState.linVelocity = JNumber3D.getScaleVector(this._currState.linVelocity, scale);
971 		this._currState.rotVelocity = JNumber3D.getScaleVector(this._currState.rotVelocity, scale);
972 	};
