 32 	 * @author Muzer(muzerly@gmail.com)

 39 	 * @requires JWheel

 42 	 * @property {number} _steerRate the rate of steering angle change
 43 	 * @property {number} _driveTorque the maximum torque of a wheel
 44 	 * @property {number} _destSteering the target steering angle
 45 	 * @property {number} _destAccelerate the target acceleration
 46 	 * @property {number} _steering the current steering angle
 47 	 * @property {number} _accelerate the current acceleration
 48 	 * @property {number} _HBrake whether the handbrake is on (1) or off (0)
 49 	 * @property {JChassis} _chassis the vehicle chassis

 79 	 * @function setCar sets up the vehicle
 80 	 * @param {number} maxSteerAngle the maximum steering angle
 81 	 * @param {number} steerRate the rate of steering angle change
 82 	 * @param {number} driveTorque the maximum torque of a wheel
 83 	 * @type void
 84 	 **/
 85 	JCar.prototype.setCar=function(maxSteerAngle, steerRate, driveTorque){
 86 		if(maxSteerAngle==null) maxSteerAngle=45;
 87 		if(steerRate==null) steerRate=4;
 88 		if(driveTorque==null) driveTorque=500;
 89 		
 90 		this._maxSteerAngle = maxSteerAngle;
 91 		this._steerRate = steerRate;
 92 		this._driveTorque = driveTorque;
 93 	};

 95 	/**
 96 	 * @function setupWheel add a wheel to the vehicle
 97 	 * @param {string} _name a unique name by which to identify the wheel (e.g. FrontRight, RearLeft etc.)
 98 	 * @param {array} pos position of the wheel relative to the car's center
 99 	 * @param {number} wheelSideFriction side friction
100 	 * @param {number} wheelFwdFriction forward friction
101 	 * @param {number} wheelTravel vertical suspension travel
102 	 * @param {number} wheelRadius wheel radius
103 	 * @param {number} wheelRestingFrac elasticity coefficient
104 	 * @param {number} wheelDampingFrac suspension damping
105 	 * @param {number} wheelNumRays 
106 	 * @param {number} drive
107 	 * @type void
108 	 **/
109 	JCar.prototype.setupWheel=function(_name, pos, wheelSideFriction, wheelFwdFriction, wheelTravel, wheelRadius, wheelRestingFrac, wheelDampingFrac, wheelNumRays, drive){
110 		if(wheelSideFriction==null) wheelSideFriction=2;
111 		if(wheelFwdFriction==null) wheelFwdFriction=2;
112 		if(wheelTravel==null) wheelTravel=3;
113 		if(wheelRadius==null) wheelRadius=10;
114 		if(wheelRestingFrac==null) wheelRestingFrac=0.5;
115 		if(wheelDampingFrac==null) wheelDampingFrac=0.5;
116 		if(wheelNumRays==null) wheelNumRays=1;
117 		if(drive==null) drive=1;
118 
119 		
120 		var gravity = PhysicsSystem.getInstance().get_gravity().slice(0);
121 		var mass = this._chassis.get_mass();
122 		var mass4 = 0.25 * mass;
123 		var gravityLen = Vector3DUtil.get_length(gravity);
124 
125 		Vector3DUtil.normalize(gravity);
126 		var axis = JNumber3D.getScaleVector(gravity,-1);
127 		var spring = mass4 * gravityLen / (wheelRestingFrac * wheelTravel);
128 		var inertia = 0.015 * wheelRadius * wheelRadius * mass;
129 		var damping = wheelDampingFrac/2;
130 		var normalForce = gravityLen*mass4;
131 		//var damping = 2 * mrSqrt(spring * mass);
132 		//damping *= (0.25 * wheelDampingFrac);
133 		//damping /= this._steerRate;
134 		//damping *= wheelDampingFrac;
135 
136 		var wheel = new JWheel(this);
137 		wheel.name = _name;
138 		wheel.setup(pos, axis, spring, wheelTravel, inertia, wheelRadius, wheelSideFriction, wheelFwdFriction, damping, wheelNumRays, drive, normalForce);
139 		this._wheels.push(wheel);
140 	};

153 	 * @param {array} wheels a collection of Wheel objects
154 	 * @param {number} val the target acceleration
155 	 * @type void
156 	 **/
157 	JCar.prototype.setSteer=function(wheels, val){
158 		this._destSteering = val;
159 		this._steerWheels = [];
160 		var wheel=null;
161 		for (var i=0, l=wheels.length; i<l; i++){
162 			wheel=this.getWheel(wheels[i]);
163 			if (wheel)
164 				this._steerWheels.push(wheel);
165 		}
166 	};

169 	 * @function findWheel checks if a wheel exists matching a given name
170 	 * @param {string} name the name of the wheel to find
171 	 * @type boolean
172 	 **/
173 	JCar.prototype.findWheel=function(_name){
174 		for (var i=0, l=this._wheels.length; i<l; i++){
175 			if (this._wheels[i].name == _name) return true;
176 		}
177 		return false;
178 	};

186 		for (var i=0; i<this._wheels.length; i++){

212 	/**
213 	 * @function postPhysics runs after the PhysicsSystem has been applied
214 	 * @param {number} dt a UNIX timestamp
215 	 * @type void
216 	 **/
217 	JCar.prototype.postPhysics=function(dt){
218 		for(var i=0, wl=this._wheels.length; i<wl; i++){
219 			this._wheels[i].update(dt);
220 		}
221 
222 		var deltaAccelerate = dt;
223 		var deltaSteering = dt * this._steerRate;
224 		var dAccelerate = this._destAccelerate - this._accelerate;
225 
226 		if (dAccelerate < -deltaAccelerate) dAccelerate = -deltaAccelerate;
227 		else if (dAccelerate > deltaAccelerate) dAccelerate = deltaAccelerate;
228 
229 		this._accelerate += dAccelerate;
230 
231 		var dSteering = this._destSteering - this._steering;
232 
233 		if (dSteering < -deltaSteering) dSteering = -deltaSteering;
234 		else if (dSteering > deltaSteering) dSteering = deltaSteering;
235 
236 		this._steering += dSteering;
237 
238 		for(var i=0;i<this._wheels.length;i++){
239 			this._wheels[i].addTorque(this._driveTorque * this._accelerate);
240 			this._wheels[i].setLock(this._HBrake > 0.5);
241 		}
242 
243 		var alpha = mrAbs(this._maxSteerAngle * this._steering);
244 		var angleSgn = (this._steering > 0) ? 1 : -1;
245 		for(var i=0, swl=this._steerWheels.length; i<swl; i++){
246 			this._steerWheels[i].setSteerAngle(angleSgn * alpha);
247 		}
248 	};

258 			//this._wheels[i].update(dt);
259 			if (this._wheels[i].getOnFloor()) count++;
260 		}
261 		return count;
