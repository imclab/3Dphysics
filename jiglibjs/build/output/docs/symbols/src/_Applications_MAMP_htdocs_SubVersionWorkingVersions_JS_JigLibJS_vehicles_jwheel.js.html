141 	 * @param {array} pos position relative to car, in car's space
142 	 * @param {array} axisUp in car's space
143 	 * @param {number} spring force per suspension offset
144 	 * @param {number} travel suspension travel upwards
145 	 * @param {number} inertia inertia about the axle
146 	 * @param {number} radius wheel radius
147 	 * @param {number} sideFriction side friction
148 	 * @param {number} fwdFriction forward friction
149 	 * @param {number} damping suspension damping
150 	 * @param {number} numRays 
151 	 * @param {number} drive

153 	 * @type void

155 	JWheel.prototype.setup=function(pos, axisUp, spring, travel, inertia, radius, sideFriction, fwdFriction, damping, numRays, drive, normalForce){
156 		if(spring==null) spring=0;
157 		if(travel==null) travel=0;
158 		if(inertia==null) inertia=0;
159 		if(radius==null) radius=0;
160 		if(sideFriction==null) sideFriction=0;
161 		if(fwdFriction==null) fwdFriction=0;
162 		if(damping==null) damping=0;
163 		if(numRays==null) numRays=0;
164 		if(drive==null) drive=0;
165 		if(normalForce==null) normalForce=0;
166 		
167 		this._pos = pos;
168 		this._axisUp = axisUp;
169 		this._spring = spring;
170 		this._travel = travel;
171 		this._inertia = inertia;
172 		this._radius = radius;
173 		this._sideFriction = sideFriction;
174 		this._fwdFriction = fwdFriction;
175 		this._damping = damping;
176 		this._numRays = numRays;
177 		this._drive = drive;
178 		this._normalForce = normalForce;
179 		this._torque = 0;
180 		this.reset();
181 	};

301 	/**
302 	 * @function addForcesToCar adds the forces from this wheel to the parent vehicle.
303 	 * @returns true if the wheel is on the ground, else false
304 	 * @type boolean
305 	 **/
306 	JWheel.prototype.addForcesToCar=function(dt){
307 		var force = [0,0,0,0];
308 		this._lastDisplacement = this._displacement;
309 		this._displacement = 0;
310 
311 		var carBody = this._car._chassis;
312 		worldPos = this._pos.slice(0);
313 		JMatrix3D.multiplyVector(carBody.get_currentState().get_orientation(), worldPos);
314 		worldPos = Vector3DUtil.add(carBody.get_currentState().position, worldPos);
315 		worldAxis = this._axisUp.slice(0);
316 		JMatrix3D.multiplyVector(carBody.get_currentState().get_orientation(), worldAxis);
317 
318 		wheelFwd = carBody.get_currentState().getOrientationCols()[2].slice(0);
319 		JMatrix3D.multiplyVector(JMatrix3D.getRotationMatrix(worldAxis[0], worldAxis[1], worldAxis[2], this._steerAngle/180*2*Math.PI), wheelFwd);
320 		wheelUp = worldAxis;
321 		wheelLeft = Vector3DUtil.crossProduct(wheelUp, wheelFwd);
322 		Vector3DUtil.normalize(wheelLeft);
323 
324 		var rayLen = 2 * this._radius + this._travel;
325 		wheelRayEnd = Vector3DUtil.subtract(worldPos, JNumber3D.getScaleVector(worldAxis, this._radius));
326 		wheelRay = new JSegment(Vector3DUtil.add(wheelRayEnd, JNumber3D.getScaleVector(worldAxis, rayLen)), JNumber3D.getScaleVector(worldAxis, -rayLen));
327 
328 		if (this._collSystem == null)
329 			this._collSystem = PhysicsSystem.getInstance().getCollisionSystem();
330 
331 		var maxNumRays = 10;
332 		var numRays = mrMin(this._numRays, maxNumRays);
333 
334 		var objArr = [];
335 		var segments = [];
336 
337 		var deltaFwd = (2 * this._radius) / (numRays +1);
338 		var deltaFwdStart = deltaFwd;
339 
340 		this._lastOnFloor = false;
341 
342 		var distFwd;
343 		var yOffset;
344 		var bestIRay = 0;
345 		var iRay = 0;
346 		var segment = null;
347 		for (iRay = 0; iRay < numRays; iRay++){
348 			objArr[iRay] = {};
349 			distFwd = (deltaFwdStart + iRay * deltaFwd) - this._radius;
350 			yOffset = this._radius * (1 - mrCos(90 * (distFwd / this._radius) * mrPI / 180));
351 			segment = wheelRay.clone();
352 			segment.origin = Vector3DUtil.add(segment.origin, Vector3DUtil.add(JNumber3D.getScaleVector(wheelFwd, distFwd), JNumber3D.getScaleVector(wheelUp, yOffset)));
353 			
354 			if (this._collSystem.segmentIntersect(objArr[iRay], segment, carBody)) {
355 				this._lastOnFloor = true;
356 				if (objArr[iRay].fracOut < objArr[bestIRay].fracOut){
357 					bestIRay = iRay;
358 				}
359 			}
360 			segments[iRay] = segment;
361 		}
362 		
363 		if (!this._lastOnFloor) return false;
364 
365 		var frac= objArr[bestIRay].fracOut;
366 		var groundPos = objArr[bestIRay].posOut;
367 		var otherBody = objArr[bestIRay].bodyOut;
368 
369 		var groundNormal = worldAxis.slice(0);
370 		if (numRays > 1){
371 			for (iRay = 0; iRay < numRays; iRay++){
372 				var rayFracOut=objArr[iRay].fracOut;
373 				if (rayFracOut <= 1)
374 					groundNormal = Vector3DUtil.add(groundNormal, JNumber3D.getScaleVector(Vector3DUtil.subtract(worldPos, segments[iRay].getEnd()), 1 - rayFracOut));
375 			}
376 			Vector3DUtil.normalize(groundNormal);
377 		}else groundNormal = objArr[bestIRay].normalOut;
378 		
379 		wheelFwd=Vector3DUtil.crossProduct(wheelLeft,groundNormal);
380 		
381 		this._displacement = rayLen * (1 - frac);
382 
383 		if (this._displacement < 0) this._displacement = 0;
384 		
385 		var mass = carBody.get_mass();
386 		var mass4 = mass/4;
387 		var otherFriction=otherBody.get_friction();
388 	
389 		var wheelCenterVel=carBody.getVelocity(this._pos);
390 		
391 		
392 		//hit floor hard
393 		var origDisplacement=this._displacement;
394 		if (this._displacement > this._travel){
395 			this._displacement=this._travel;
396 			var cv=Vector3DUtil.dotProduct(wheelCenterVel,groundNormal)/dt*mass4;
397 			cv=cv*2*otherBody.get_restitution()/10;
398 			extraForce = JNumber3D.getScaleVector(groundNormal, -cv);
399 			force = Vector3DUtil.add(force, extraForce);
400 		}
401 		//suspension spring force
402 		extraForce = JNumber3D.getScaleVector(this._axisUp, this._spring*this._displacement+this._upSpeed*this._damping);
403 		force = Vector3DUtil.add(force, extraForce);
404 
405 		
406 		groundUp = groundNormal;
407 		groundLeft = Vector3DUtil.crossProduct(groundNormal, wheelFwd);
408 		Vector3DUtil.normalize(groundLeft);
409 		groundFwd = Vector3DUtil.crossProduct(groundLeft, groundUp);
410 	
411 		var rimVel = JNumber3D.getScaleVector(Vector3DUtil.crossProduct(groundLeft, Vector3DUtil.subtract(groundPos, worldPos)), -this._angVel);
412 		var centerVel = JNumber3D.getScaleVector(groundFwd, Vector3DUtil.dotProduct(wheelCenterVel,groundFwd));
413 
414 		var friction=this._fwdFriction*otherFriction;
415 		var extraForce = JNumber3D.getScaleVector(Vector3DUtil.subtract(rimVel,centerVel), mass4/dt/this._radius*friction);
416 		var forceSize=Vector3DUtil.get_length(extraForce);
417 		if(forceSize>this._normalForce*friction) extraForce = JNumber3D.getScaleVector(extraForce,this._normalForce*friction/forceSize);
418 		force = Vector3DUtil.add(force, extraForce);		
419 		this._torque-=Vector3DUtil.dotProduct(Vector3DUtil.subtract(rimVel,centerVel),groundFwd)/this._radius*mass4/dt;
420 		this._angVelForGrip = Vector3DUtil.dotProduct(wheelCenterVel, groundFwd) / this._radius;
421 
422 		//sideways friction
423 		var sideVel = Vector3DUtil.dotProduct(wheelCenterVel,groundLeft);
424 		var friction=this._sideFriction*otherFriction;		
425 		var leftVel = JNumber3D.getScaleVector(groundLeft, -sideVel*friction);
426 
427 		
428 		var extraForce = JNumber3D.getScaleVector(leftVel, mass4/dt/this._radius);	
429 		var forceSize=Vector3DUtil.get_length(extraForce);
430 		if(forceSize>this._normalForce*friction) extraForce = JNumber3D.getScaleVector(extraForce,this._normalForce*friction/forceSize);
431 		
432 		force = Vector3DUtil.add(force, extraForce);
433 
434 		carBody.addWorldForce(force, groundPos);
435 		if (otherBody.get_movable()){
436 			var maxOtherBodyAcc = 500;
437 			var maxOtherBodyForce = maxOtherBodyAcc * otherBody.get_mass();
438 			if (Vector3DUtil.get_lengthSquared(force) > (maxOtherBodyForce * maxOtherBodyForce))
439 				force = JNumber3D.getScaleVector(force, maxOtherBodyForce / Vector3DUtil.get_length(force));
440 
441 			otherBody.addWorldForce(JNumber3D.getScaleVector(force, -1), groundPos);
442 		}
443 		return true;
444 	};

446 	/**
447 	 * @function update updates the rotational state etc
448 	 * @type void
449 	 **/
450 	JWheel.prototype.update=function(dt){
451 		if (dt <= 0) return;
452 
453 		var origAngVel = this._angVel;
454 		this._upSpeed = (this._displacement - this._lastDisplacement) / mrMax(dt, JNumber3D.NUM_TINY);
455 
456 		if (this._locked){
457 			this._angVel = 0;
458 			this._torque = 0;
459 		}else{					
460 			this._angVel += (this._torque * dt / this._inertia);
461 			this._torque = 0;
462 			
463 			if (((origAngVel > this._angVelForGrip) && (this._angVel < this._angVelForGrip)) || ((origAngVel < this._angVelForGrip) && (this._angVel > this._angVelForGrip)))
464 				this._angVel = this._angVelForGrip;
465 			
466 			
467 			this._angVel += this._driveTorque * dt / this._inertia * this._drive;
468 			this._driveTorque = 0;
469 
470 			if (this._angVel < -100) this._angVel = -100;
471 			else if (this._angVel > 100) this._angVel = 100;
472 			
473 			this._angVel *= this._rotDamping;
474 			this._axisAngle += (this._angVel * dt * 180 / mrPI);
475 		}
476 	};

483 		this._angVel = 0;
484 		this._steerAngle = 0;
